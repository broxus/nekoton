30    pub fn parse<'tx>(&'tx self, tx: &'tx ton_block::Transaction) -> Result<Vec<Extracted<'tx>>> {
31        let mut output = Vec::new();
32
33        if let Some(message) = &tx.in_msg {
34            let message_hash = message.hash();
35            let message = message.read_struct().context("Failed reading in msg")?;
36
37            if let Some(body) = message.body() {
38                output.extend(
39                    self.parse_in_message(&message, body)?
40                        .map(|parsed| Extracted {
41                            function_id: parsed.function_id,
42                            name: parsed.name,
43                            bounced: parsed.bounced,
44                            tokens: parsed.tokens,
45                            message_hash,
46                            message,
47                            tx,
48                            is_in_message: true,
49                            index_in_transaction: 0,
50                            parsed_type: ParsedType::FunctionInput,
51                            decoded_headers: parsed.headers,
52                        }),
53                );
54            }
55        }
56
57        let mut index_in_transaction = 0;
58        tx.out_msgs.iterate_slices(|slice| {
59            let message = slice.reference(0)?;
60            let message_hash = message.repr_hash();
61            let message = ton_block::Message::construct_from_cell(message)?;
62
63            if let Some(body) = message.body() {
64                let function_id = read_function_id(&body)?;
65                output.extend(
66                    self.parse_out_message(message.header(), function_id, body)?
67                        .map(|parsed| Extracted {
68                            function_id,
69                            name: parsed.name,
70                            bounced: false,
71                            tokens: parsed.tokens,
72                            message_hash,
73                            message,
74                            tx,
75                            is_in_message: false,
76                            index_in_transaction,
77                            parsed_type: parsed.parsed_type,
78                            decoded_headers: Default::default(),
79                        }),
80                );
81            }
82
83            index_in_transaction += 1;
84            Ok(true)
85        })?;
86
87        Ok(output)
88    }

90    fn parse_in_message(
91        &self,
92        message: &ton_block::Message,
93        mut body: SliceData,
94    ) -> Result<Option<ParsedInMessage<'_>>> {
95        if matches!(
96            message.header(), ton_block::CommonMsgInfo::ExtInMsgInfo(_)
97            if self.match_external_in
98        ) {
99            let (headers, function_id, body) =
100                ton_abi::Function::decode_header(&self.abi_version, body, &self.headers, false)?;
101
102            return if let Some(fun) = self.functions.get(&function_id) {
103                let tokens = TokenValue::decode_params(
104                    fun.function.input_params(),
105                    body.slice,
106                    &self.abi_version,
107                    false,
108                )?;
109
110                Ok(Some(ParsedInMessage {
111                    function_id,
112                    name: &fun.function.name,
113                    bounced: false,
114                    tokens,
115                    headers,
116                }))
117            } else {
118                Ok(None)
119            };
120        }
121
122        if message.bounced() {
123            body.get_next_u32()?; //skip bounce bytes
124            let function_id = read_function_id(&body)?;
125            if let Some(abi) = self.functions_with_bounce.get(&function_id) {
126                return Ok(Some(ParsedInMessage {
127                    name: &abi.function.name,
128                    function_id,
129                    headers: Default::default(),
130                    tokens: (abi.bounce_handler)(body)?,
131                    bounced: true,
132                }));
133            }
134        }
135
136        let function_id = read_function_id(&body)?;
137        match self.functions.get(&function_id) {
138            Some(abi) => Ok(Some(ParsedInMessage {
139                name: &abi.function.name,
140                function_id,
141                headers: Default::default(),
142                tokens: abi
143                    .decode_input(body, message.is_internal())
144                    .context("Failed decoding input message")?,
145                bounced: false,
146            })),
147            None => Ok(None),
148        }
149    }

151    fn parse_out_message(
152        &self,
153        info: &ton_block::CommonMsgInfo,
154        function_id: u32,
155        body: SliceData,
156    ) -> Result<Option<ParsedOutMessage<'_>>> {
157        let is_internal = match info {
158            ton_block::CommonMsgInfo::ExtOutMsgInfo(_) => {
159                if let Some(event) = self.events.get(&function_id) {
160                    return Ok(Some(ParsedOutMessage {
161                        name: &event.name,
162                        parsed_type: ParsedType::Event,
163                        tokens: event.decode_input(body)?,
164                    }));
165                }
166
167                false
168            }
169            ton_block::CommonMsgInfo::IntMsgInfo(_) => true,
170            ton_block::CommonMsgInfo::ExtInMsgInfo(_) => return Ok(None),
171        };
172
173        Ok(if let Some(abi) = self.functions.get(&function_id) {
174            let (parsed_type, tokens) = if abi.function.input_id == function_id {
175                (
176                    ParsedType::FunctionInput,
177                    abi.decode_input(body, is_internal),
178                )
179            } else {
180                (
181                    ParsedType::FunctionOutput,
182                    abi.decode_output(body, is_internal),
183                )
184            };
185
186            Some(ParsedOutMessage {
187                name: &abi.function.name,
188                parsed_type,
189                tokens: tokens.context("Failed decoding output message")?,
190            })
191        } else {
192            None
193        })
194    }

294    /// Returns `Err` if there are duplicate `function_ids`
295    /// # Params:
296    /// * `match_external_in` - if true, `ExternalIn` messages will be matched. Not compatible with several different contracts.
297    fn build_impl(self, match_external_in: bool) -> Result<TransactionParser> {
298        let mut abi_version = None;
299        let mut headers = Vec::new();
300
301        let mut functions = FxHashMap::default();
302        let mut functions_with_bounce = FxHashMap::default();
303        let mut events = FxHashMap::default();
304
305        let mut set_abi_version = |new_version| match abi_version {
306            Some(version) if version == new_version => Ok(()),
307            Some(_) if match_external_in => anyhow::bail!("Multiple different ABI versions"),
308            _ => {
309                abi_version = Some(new_version);
310                Ok(())
311            }
312        };
313        let mut set_headers = |new_headers: &mut Vec<ton_abi::Param>| {
314            if !match_external_in {
315                return Ok(());
316            }
317
318            if headers.is_empty() {
319                headers = new_headers.clone();
320                Ok(())
321            } else if headers.len() != new_headers.len()
322                || headers
323                    .iter()
324                    .zip(new_headers.iter())
325                    .any(|(a, b)| a.name != b.name)
326            {
327                Err(anyhow::anyhow!("Different headers sets are not supported"))
328            } else {
329                Ok(())
330            }
331        };
332
333        for mut abi in self.functions_in {
334            set_abi_version(abi.function.abi_version)?;
335            set_headers(&mut abi.function.header)?;
336
337            match functions.entry(abi.function.input_id) {
338                hash_map::Entry::Vacant(entry) => {
339                    entry.insert(abi);
340                }
341                hash_map::Entry::Occupied(entry) => {
342                    let WrappedFunction { function, .. } = entry.get();
343                    anyhow::bail!(
344                        "Duplicate function id for IN function. Id: {}. Name: {}",
345                        function.input_id,
346                        function.name,
347                    )
348                }
349            }
350        }
351
352        for mut abi in self.functions_out {
353            set_abi_version(abi.function.abi_version)?;
354            set_headers(&mut abi.function.header)?;
355
356            match functions.entry(abi.function.output_id) {
357                hash_map::Entry::Vacant(entry) => {
358                    entry.insert(abi);
359                }
360                hash_map::Entry::Occupied(entry) => {
361                    let WrappedFunction { function, .. } = entry.get();
362                    if function.input_id == function.output_id {
363                        log::warn!(
364                            "Function with same input and output id: {}. \
365                            Adding it only as IN function parser",
366                            abi.function.name
367                        );
368                        continue;
369                    }
370                    anyhow::bail!(
371                        "Duplicate function id for OUT function. Id: {}. Name: {}",
372                        function.output_id,
373                        function.name
374                    )
375                }
376            }
377        }
378
379        for abi in self.functions_with_bounce {
380            set_abi_version(abi.function.abi_version)?;
381
382            match functions_with_bounce.entry(abi.function.input_id) {
383                hash_map::Entry::Vacant(entry) => {
384                    entry.insert(abi);
385                }
386                hash_map::Entry::Occupied(entry) => {
387                    let FunctionWithBounceHandler { function, .. } = entry.get();
388                    anyhow::bail!(
389                        "Duplicate function id for function with bounce. Id: {}. Name: {}",
390                        function.input_id,
391                        function.name
392                    )
393                }
394            }
395        }
396
397        for event in self.events {
398            set_abi_version(event.abi_version)?;
399
400            match events.entry(event.id) {
401                hash_map::Entry::Vacant(entry) => {
402                    entry.insert(event);
403                }
404                hash_map::Entry::Occupied(entry) => {
405                    let event = entry.get();
406                    anyhow::bail!("Duplicate event id. Id: {}. Name: {}", event.id, event.name);
407                }
408            }
409        }
410
411        Ok(TransactionParser {
412            abi_version: abi_version.context("Matching functions list is empty")?,
413            headers,
414            functions,
415            functions_with_bounce,
416            events,
417            match_external_in,
418        })
419    }
