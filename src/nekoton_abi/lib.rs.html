107pub fn read_function_id(data: &SliceData) -> Result<u32> {
108    let mut value: u32 = 0;
109    for i in 0..4 {
110        value |= (data.get_byte(8 * i)? as u32) << (8 * (3 - i));
111    }
112    Ok(value)
113}

115pub fn read_input_function_id(
116    contract: &ton_abi::Contract,
117    mut body: SliceData,
118    internal: bool,
119) -> Result<u32> {
120    if !internal {
121        // Skip optional signature
122        if body.get_next_bit()? {
123            body.move_by(ed25519_dalek::SIGNATURE_LENGTH * 8)?;
124        }
125
126        // Skip headers
127        for header in &contract.header {
128            match header.kind {
129                ton_abi::ParamType::PublicKey => {
130                    if body.get_next_bit()? {
131                        body.move_by(ed25519_dalek::PUBLIC_KEY_LENGTH * 8)?;
132                    }
133                }
134                ton_abi::ParamType::Time => body.move_by(64)?,
135                ton_abi::ParamType::Expire => body.move_by(32)?,
136                _ => return Err(AbiError::UnsupportedHeader.into()),
137            }
138        }
139    }
140
141    read_function_id(&body)
142}

144pub fn guess_method_by_input<'a>(
145    contract: &'a ton_abi::Contract,
146    message_body: &SliceData,
147    method: &MethodName,
148    internal: bool,
149) -> Result<Option<&'a ton_abi::Function>> {
150    let names = match method {
151        MethodName::Known(name) => return Ok(Some(contract.function(name)?)),
152        MethodName::GuessInRange(names) => Some(names),
153        MethodName::Guess => None,
154    };
155
156    let input_id = match read_input_function_id(contract, message_body.clone(), internal) {
157        Ok(id) => id,
158        Err(_) => return Ok(None),
159    };
160
161    let mut method = None;
162    match names {
163        Some(names) => {
164            for name in names {
165                let function = contract.function(name)?;
166                if function.input_id == input_id {
167                    method = Some(function);
168                    break;
169                }
170            }
171        }
172        None => {
173            for function in contract.functions.values() {
174                if function.input_id == input_id {
175                    method = Some(function);
176                    break;
177                }
178            }
179        }
180    }
181    Ok(method)
182}

208pub fn parse_comment_payload(mut payload: SliceData) -> Option<String> {
209    if payload.get_next_u32().ok()? != 0 {
210        return None;
211    }
212
213    let mut cell = payload.checked_drain_reference().ok()?;
214
215    let mut data = Vec::new();
216    loop {
217        data.extend_from_slice(cell.data());
218        cell = match cell.reference(0) {
219            Ok(cell) => cell.clone(),
220            Err(_) => break,
221        };
222    }
223
224    String::from_utf8(data).ok()
225}

262pub fn extract_public_key(
263    account: &AccountStuff,
264) -> Result<ed25519_dalek::PublicKey, ExtractionError> {
265    let state_init = match &account.storage.state {
266        ton_block::AccountState::AccountActive { state_init, .. } => state_init,
267        _ => return Err(ExtractionError::AccountIsNotActive),
268    };
269    let mut data: SliceData = match &state_init.data {
270        Some(data) => SliceData::load_cell_ref(data).map_err(|_| ExtractionError::CellUnderflow)?,
271        None => return Err(ExtractionError::AccountDataNotFound),
272    };
273    let data = data
274        .get_next_bytes(32)
275        .map_err(|_| ExtractionError::CellUnderflow)?;
276
277    ed25519_dalek::PublicKey::from_bytes(&data).map_err(|_| ExtractionError::InvalidPublicKey)
278}

297pub fn insert_state_init_data(
298    contract: &ton_abi::Contract,
299    data: SliceData,
300    public_key: &Option<ed25519_dalek::PublicKey>,
301    tokens: Vec<Token>,
302) -> Result<SliceData> {
303    #[derive(thiserror::Error, Debug)]
304    enum InitDataError {
305        #[error("Token not found: {}", .0)]
306        TokenNotFound(String),
307        #[error("Token param type mismatch")]
308        TokenParamTypeMismatch,
309    }
310
311    if contract.data_map_supported() {
312        let mut map = ton_types::HashmapE::with_hashmap(
313            ton_abi::Contract::DATA_MAP_KEYLEN,
314            data.reference_opt(0),
315        );
316
317        if let Some(public_key) = public_key {
318            map.set_builder(
319                0u64.serialize().and_then(SliceData::load_cell).trust_me(),
320                ton_types::BuilderData::new()
321                    .append_raw(public_key.as_bytes(), 256)
322                    .trust_me(),
323            )?;
324        }
325
326        if !contract.data.is_empty() {
327            let tokens = tokens
328                .into_iter()
329                .map(|token| (token.name, token.value))
330                .collect::<HashMap<_, _>>();
331
332            for (param_name, param) in &contract.data {
333                let token = tokens
334                    .get(param_name)
335                    .ok_or_else(|| InitDataError::TokenNotFound(param_name.clone()))?;
336                if !token.type_check(&param.value.kind) {
337                    return Err(InitDataError::TokenParamTypeMismatch.into());
338                }
339
340                let key = param.key.serialize();
341                let key = key.and_then(SliceData::load_cell).trust_me();
342
343                let builder = token.pack_into_chain(&contract.abi_version)?;
344                map.set_builder(key, &builder)?;
345            }
346        }
347
348        return map.write_to_new_cell().and_then(SliceData::load_builder);
349    }
350
351    if contract.init_fields_supported() {
352        let mut init_fields = HashMap::with_capacity(tokens.capacity());
353        for init_field_name in &contract.init_fields {
354            match (init_field_name.as_str(), public_key) {
355                ("_pubkey", Some(public_key)) => {
356                    init_fields.insert(
357                        init_field_name.to_string(),
358                        UInt256::with_array(*public_key.as_bytes()).token_value(),
359                    );
360                }
361                (name, _) => {
362                    if let Some(token) = tokens.iter().find(|x| x.name == name) {
363                        init_fields.insert(name.to_string(), token.value.clone());
364                    } else {
365                        return Err(InitDataError::TokenNotFound(name.to_string()).into());
366                    }
367                }
368            }
369        }
370
371        let builder_data = contract.encode_storage_fields(init_fields)?;
372        return SliceData::load_builder(builder_data);
373    }
374
375    SliceData::load_cell(ton_types::Cell::default())
376}

393pub fn decode_output<'a>(
394    contract: &'a ton_abi::Contract,
395    message_body: SliceData,
396    method: &MethodName,
397) -> Result<Option<(&'a Function, Vec<Token>)>> {
398    let output_id = match read_function_id(&message_body) {
399        Ok(id) => id,
400        Err(_) => return Ok(None),
401    };
402
403    let function = match method {
404        MethodName::Known(name) => Some(contract.function(name)?),
405        MethodName::GuessInRange(names) => {
406            let mut function = None;
407            for name in names {
408                let entry = contract.function(name)?;
409                if entry.output_id == output_id {
410                    function = Some(entry);
411                    break;
412                }
413            }
414            function
415        }
416        MethodName::Guess => {
417            let mut function = None;
418            for entry in contract.functions.values() {
419                if entry.output_id == output_id {
420                    function = Some(entry);
421                    break;
422                }
423            }
424            function
425        }
426    };
427
428    let function = match function {
429        Some(function) => function,
430        None => return Ok(None),
431    };
432
433    let output = function.decode_output(message_body, true)?;
434    Ok(Some((function, output)))
435}

437pub fn decode_event<'a>(
438    contract: &'a ton_abi::Contract,
439    message_body: SliceData,
440    name: &MethodName,
441) -> Result<Option<(&'a ton_abi::Event, Vec<Token>)>> {
442    let events = &contract.events;
443    let event_id = match read_function_id(&message_body) {
444        Ok(id) => id,
445        Err(_) => return Ok(None),
446    };
447
448    let event = match name {
449        MethodName::Known(name) => events.get(name),
450        MethodName::GuessInRange(names) => {
451            let mut event = None;
452            for name in names {
453                let entry = match events.get(name) {
454                    Some(event) => event,
455                    None => continue,
456                };
457                if entry.id == event_id {
458                    event = Some(entry);
459                    break;
460                }
461            }
462            event
463        }
464        MethodName::Guess => {
465            let mut event = None;
466            for entry in events.values() {
467                if entry.id == event_id {
468                    event = Some(entry);
469                    break;
470                }
471            }
472            event
473        }
474    };
475
476    let event = match event {
477        Some(event) => event,
478        None => return Ok(None),
479    };
480
481    let data = event.decode_input(message_body)?;
482    Ok(Some((event, data)))
483}

581
582    Ok(ton_block::StateInit {
583        code: Some(code),
584        data: Some(data),
585        ..Default::default()
586    })

764impl<'a> FunctionAbi<'a> {
765    fn new(abi: &'a Function) -> Self {
766        Self { abi }
767    }

773
774    fn run_local(
775        &self,
776        clock: &dyn Clock,
777        account_stuff: &mut AccountStuff,
778        input: &[Token],
779        responsible: bool,
780        config: &BriefBlockchainConfig,
781    ) -> Result<ExecutionOutput> {
782        let function = self.abi;
783
784        let answer_id = if responsible {
785            account_stuff.storage.balance.grams = ton_block::Grams::from(100_000_000_000_000u64); // 100 000 TON
786
787            match input.first().map(|token| &token.value) {
788                Some(TokenValue::Uint(ton_abi::Uint { number, size: 32 })) => number
789                    .to_u32()
790                    .map(Some)
791                    .ok_or(AbiError::AnswerIdNotFound)?,
792                _ => return Err(AbiError::AnswerIdNotFound.into()),
793            }
794        } else {
795            None
796        };
797
798        let mut msg = if responsible {
799            ton_block::Message::with_int_header(ton_block::InternalMessageHeader {
800                src: ton_block::MsgAddressIntOrNone::Some(account_stuff.addr.clone()),
801                dst: account_stuff.addr.clone(),
802                ..Default::default()
803            })
804        } else {
805            ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
806                dst: account_stuff.addr.clone(),
807                ..Default::default()
808            })
809        };
810
811        let BlockStats {
812            now_ms,
813            gen_utime,
814            gen_lt,
815        } = get_block_stats(clock, None, account_stuff.storage.last_trans_lt);
816
817        if responsible {
818            msg.set_body(
819                function
820                    .encode_internal_input(input)
821                    .and_then(SliceData::load_builder)?,
822            );
823        } else {
824            msg.set_body(
825                self.abi
826                    .encode_run_local_input(now_ms, input)
827                    .and_then(SliceData::load_builder)?,
828            );
829        }
830
831        let tvm::ActionPhaseOutput {
832            messages,
833            exit_code: result_code,
834        } = tvm::call_msg(
835            gen_utime,
836            gen_lt,
837            account_stuff,
838            &msg,
839            config,
840            &Default::default(),
841        )?;
842
843        let tokens = if let Some(answer_id) = answer_id {
844            messages.map(|messages| {
845                let mut output = None;
846
847                for msg in messages {
848                    if !matches!(msg.header(), ton_block::CommonMsgInfo::IntMsgInfo(_)) {
849                        continue;
850                    }
851
852                    let mut body = match msg.body() {
853                        Some(body) => body,
854                        None => continue,
855                    };
856
857                    if !matches!(
858                        body.get_next_u32(),
859                        Ok(target_answer_id) if target_answer_id == answer_id
860                    ) {
861                        continue;
862                    }
863
864                    if let Ok(tokens) = TokenValue::decode_params(
865                        function.output_params(),
866                        body,
867                        &function.abi_version,
868                        false,
869                    ) {
870                        output = Some(tokens);
871                        break;
872                    }
873                }
874
875                match output {
876                    Some(a) => Ok(a),
877                    None if !function.has_output() => Ok(Default::default()),
878                    None => Err(AbiError::NoMessagesProduced.into()),
879                }
880            })
881        } else {
882            messages.map(|messages| process_out_messages(&messages, self.abi))
883        }
884        .transpose()?;
885
886        Ok(ExecutionOutput {
887            tokens,
888            result_code,
889        })
890    }
891}

899pub fn process_out_messages(
900    messages: &[ton_block::Message],
901    abi_function: &Function,
902) -> Result<Vec<Token>> {
903    let mut output = None;
904
905    for msg in messages {
906        if !matches!(msg.header(), ton_block::CommonMsgInfo::ExtOutMsgInfo(_)) {
907            continue;
908        }
909
910        let body = msg.body().ok_or(AbiError::InvalidOutputMessage)?;
911
912        if abi_function.is_my_output_message(body.clone(), false)? {
913            let tokens = abi_function.decode_output(body, false)?;
914
915            output = Some(tokens);
916            break;
917        }
918    }
919
920    match output {
921        Some(a) => Ok(a),
922        None if !abi_function.has_output() => Ok(Default::default()),
923        None => Err(AbiError::NoMessagesProduced.into()),
924    }
925}

927pub fn process_raw_outputs(
928    ext_out_msg_bodies: &[SliceData],
929    abi_function: &Function,
930) -> Result<Vec<Token>> {
931    let mut output = None;
932
933    for body in ext_out_msg_bodies {
934        let function_id = read_function_id(body).map_err(|_| AbiError::InvalidOutputMessage)?;
935        if abi_function.output_id != function_id {
936            continue;
937        }
938
939        output = Some(abi_function.decode_output(body.clone(), false)?);
940        break;
941    }
942
943    match output {
944        Some(a) => Ok(a),
945        None if !abi_function.has_output() => Ok(Default::default()),
946        None => Err(AbiError::NoMessagesProduced.into()),
947    }
948}

950pub fn parse_transaction_messages(
951    transaction: &ton_block::Transaction,
952) -> Result<Vec<ton_block::Message>> {
953    let mut messages = Vec::new();
954    transaction.out_msgs.iterate_slices(|slice| {
955        if let Ok(message) = slice
956            .reference(0)
957            .and_then(ton_block::Message::construct_from_cell)
958        {
959            messages.push(message);
960        }
961        Ok(true)
962    })?;
963    Ok(messages)
964}

1124            now_ms,

1126            gen_lt,
1127        },

1129            now_ms,
1130            gen_utime: (now_ms / 1000) as u32,
1131            gen_lt: last_trans_lt + UNKNOWN_TRANSACTION_LT_OFFSET,
1132        },
1133    }
