8const ANSWER_ID: &str = "_answer_id";
9
10#[derive(Debug, Clone)]
11pub struct FunctionBuilder {
12    /// Explicit function id
13    id: Option<u32>,
14    /// Contract function specification.
15    /// ABI version
16    abi_version: AbiVersion,
17    /// Function name.
18    name: String,
19    /// Function header parameters.
20    header: Vec<Param>,
21    /// Function input.
22    inputs: Vec<Param>,
23    /// Function output.
24    outputs: Vec<Param>,
25    /// Whether `answer_id` is set
26    responsible: bool,
27}
28
29impl FunctionBuilder {
30    pub fn new(function_name: &str) -> Self {
31        Self {
32            id: None,
33            abi_version: ABI_VERSION_2_0,
34            name: function_name.to_string(),
35            header: Vec::new(),
36            inputs: Vec::new(),
37            outputs: Vec::new(),
38            responsible: false,
39        }
40    }
41
42    pub fn new_responsible(function_name: &str) -> Self {
43        let mut function = Self::new(function_name);
44        function.make_responsible();
45        function
46    }
47
48    pub fn default_headers(self) -> Self {
49        self.pubkey_header().time_header().expire_header()
50    }
51
52    pub fn pubkey_header(self) -> Self {
53        self.header("pubkey", ParamType::PublicKey)
54    }
55
56    pub fn time_header(self) -> Self {
57        self.header("time", ParamType::Time)
58    }
59
60    pub fn expire_header(self) -> Self {
61        self.header("expire", ParamType::Expire)
62    }
63
64    pub fn make_responsible(&mut self) {
65        if self.inputs.is_empty() {
66            self.inputs.push(Param::new(ANSWER_ID, ParamType::Uint(32)));
67        } else if !self.responsible {
68            self.inputs
69                .insert(0, Param::new(ANSWER_ID, ParamType::Uint(32)));
70        }
71        self.responsible = true;
72    }
73
74    pub fn abi_version(mut self, abi_version: AbiVersion) -> Self {
75        self.abi_version = abi_version;
76        self
77    }
78
79    pub fn id(mut self, id: u32) -> Self {
80        self.id = Some(id);
81        self
82    }
83
84    /// Adds input param
85    #[deprecated(note = "use `input` instead")]
86    pub fn in_arg(self, name: &str, ty: ParamType) -> Self {
87        self.input(name, ty)
88    }
89
90    /// Adds input param
91    pub fn input(mut self, name: &str, ty: ParamType) -> Self {
92        self.inputs.push(Param::new(name, ty));
93        self
94    }
95
96    /// Sets the input params to the specified
97    ///
98    /// NOTE: Replaces previously added inputs
99    pub fn inputs(mut self, inputs: Vec<Param>) -> Self {
100        self.inputs = inputs;
101        self
102    }
103
104    /// Adds output param
105    #[deprecated(note = "use `output` instead")]
106    pub fn out_arg(self, name: &str, ty: ton_abi::ParamType) -> Self {
107        self.output(name, ty)
108    }
109
110    /// Adds output param
111    pub fn output(mut self, name: &str, ty: ton_abi::ParamType) -> Self {
112        self.outputs.push(Param::new(name, ty));
113        self
114    }
115
116    /// Sets the output params to the specified
117    ///
118    /// NOTE: Replaces previously added outputs
119    pub fn outputs(mut self, outputs: Vec<Param>) -> Self {
120        self.outputs = outputs;
121        self
122    }
123
124    pub fn header(mut self, name: &str, arg_type: ton_abi::ParamType) -> Self {
125        self.header.push(Param::new(name, arg_type));
126        self
127    }
128
129    pub fn headers(mut self, headers: Vec<Param>) -> Self {
130        self.header = headers;
131        self
132    }
133
134    pub fn build(self) -> Function {
135        let mut fun = Function {
136            abi_version: self.abi_version,
137            name: self.name,
138            header: self.header,
139            inputs: self.inputs,
140            outputs: self.outputs,
141            input_id: 0,
142            output_id: 0,
143        };
144        match self.id {
145            Some(id) => {
146                fun.input_id = id;
147                fun.output_id = id;
148            }
149            None => {
150                let id = fun.get_function_id();
151                fun.input_id = id & 0x7FFFFFFF;
152                fun.output_id = id | 0x80000000;
153            }
154        }
155        fun
156    }
157}

159#[derive(Default, Debug, Clone)]
160pub struct TupleBuilder {
161    types: Vec<Param>,
162}
163
164impl TupleBuilder {
165    pub fn new() -> Self {
166        Self::default()
167    }
168
169    #[deprecated(note = "use `item` instead")]
170    pub fn arg(self, name: &str, ty: ParamType) -> Self {
171        self.item(name, ty)
172    }
173
174    pub fn item(mut self, name: &str, ty: ParamType) -> Self {
175        self.types.push(Param::new(name, ty));
176        self
177    }
178
179    pub fn build(self) -> ParamType {
180        ParamType::Tuple(self.types)
181    }
182}
183
184impl FromIterator<Param> for TupleBuilder {
185    fn from_iter<T: IntoIterator<Item = Param>>(iter: T) -> Self {
186        Self {
187            types: iter.into_iter().collect(),
188        }
189    }
190}
