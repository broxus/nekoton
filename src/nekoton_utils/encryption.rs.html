13/// Decrypts utf8 data using specified decrypter and nonce
14pub fn decrypt_secure_str(
15    dec: &ChaCha20Poly1305,
16    nonce: &Nonce,
17    data: &[u8],
18) -> Result<SecUtf8, SymmetricCryptoError> {
19    String::from_utf8(decrypt(dec, nonce, data)?)
20        .map(SecUtf8::from)
21        .map_err(|e| {
22            e.into_bytes().zeroize();
23            SymmetricCryptoError::FailedToDecryptData
24        })
25}

27/// Decrypts data using specified decrypter and nonce
28pub fn decrypt_secure(
29    dec: &ChaCha20Poly1305,
30    nonce: &Nonce,
31    data: &[u8],
32) -> Result<SecVec<u8>, SymmetricCryptoError> {
33    decrypt(dec, nonce, data).map(SecVec::new)
34}
35
36/// Decrypts data using specified decrypter and nonce
37pub fn decrypt(
38    dec: &ChaCha20Poly1305,
39    nonce: &Nonce,
40    data: &[u8],
41) -> Result<Vec<u8>, SymmetricCryptoError> {
42    dec.decrypt(nonce, data)
43        .map_err(|_| SymmetricCryptoError::FailedToDecryptData)
44}
45
46/// Encrypts data using specified encryptor and nonce
47pub fn encrypt(
48    enc: &ChaCha20Poly1305,
49    nonce: &Nonce,
50    data: &[u8],
51) -> Result<Vec<u8>, SymmetricCryptoError> {
52    enc.encrypt(nonce, data)
53        .map_err(|_| SymmetricCryptoError::FailedToEncryptData)
54}

56// Calculates symmetric key from user password, using pbkdf2
57pub fn symmetric_key_from_password(password: &str, salt: &[u8]) -> Key {
58    let mut pbkdf2_hash = SecVec::new(vec![0; CREDENTIAL_LEN]);
59    pbkdf2_hmac::<sha2::Sha256>(
60        password.as_bytes(),
61        salt,
62        N_ITER,
63        pbkdf2_hash.unsecure_mut(),
64    );
65    Key::clone_from_slice(pbkdf2_hash.unsecure())
66}
