34impl<'a> Container<'a> {
35    pub fn from_ast(cx: &ParsingContext, input: &'a syn::DeriveInput) -> Option<Container<'a>> {
36        let attrs = attr::Container::from_ast(cx, input)?;
37
38        let data = match &input.data {
39            syn::Data::Enum(data) => Data::Enum(enum_from_ast(cx, &data.variants)?),
40            syn::Data::Struct(data) => {
41                let (style, fields) = struct_from_ast(cx, &data.fields)?;
42                Data::Struct(style, fields)
43            }
44            syn::Data::Union(_) => {
45                cx.error_spanned_by(input, "union types are not supported");
46                return None;
47            }
48        };
49
50        let item = Self {
51            ident: input.ident.clone(),
52            attrs,
53            data,
54            generics: &input.generics,
55            original: input,
56        };
57        // TODO: check item
58        Some(item)
59    }
60}

62impl<'a> Data<'a> {
63    #[allow(dead_code)]
64    pub fn all_fields(&'a self) -> impl Iterator<Item = &'a Field<'a>> {
65        match self {
66            Data::Enum(variants) => {
67                Either::Left(variants.iter().flat_map(|variant| variant.fields.iter()))
68            }
69            Data::Struct(_, fields) => Either::Right(fields.iter()),
70        }
71    }
72}

74fn enum_from_ast<'a>(
75    cx: &ParsingContext,
76    variants: &'a Punctuated<syn::Variant, syn::Token![,]>,
77) -> Option<Vec<Variant<'a>>> {
78    let has_consistent_discriminants = {
79        let mut iter = variants.iter();
80        match iter.next() {
81            Some(variant) => {
82                iter.all(|item| item.discriminant.is_some() == variant.discriminant.is_some())
83            }
84            None => true,
85        }
86    };
87
88    if !has_consistent_discriminants {
89        cx.error_spanned_by(variants, "variant discriminants are not consistent");
90        return None;
91    }
92
93    let result: Vec<Variant<'_>> = variants
94        .iter()
95        .flat_map(|variant| {
96            let (style, fields) = struct_from_ast(cx, &variant.fields)?;
97            Some(Variant {
98                ident: variant.ident.clone(),
99                style,
100                fields,
101                original: variant,
102            })
103        })
104        .collect();
105
106    (result.len() == variants.len()).then_some(result)
107}

109fn struct_from_ast<'a>(
110    cx: &ParsingContext,
111    fields: &'a syn::Fields,
112) -> Option<(StructStyle, Vec<Field<'a>>)> {
113    match fields {
114        syn::Fields::Named(fields) => {
115            Some((StructStyle::Struct, fields_from_ast(cx, &fields.named)?))
116        }
117        syn::Fields::Unnamed(fields) if fields.unnamed.len() == 1 => {
118            Some((StructStyle::NewType, fields_from_ast(cx, &fields.unnamed)?))
119        }
120        syn::Fields::Unnamed(fields) => {
121            Some((StructStyle::Tuple, fields_from_ast(cx, &fields.unnamed)?))
122        }
123        syn::Fields::Unit => Some((StructStyle::Unit, Vec::new())),
124    }
125}

127fn fields_from_ast<'a>(
128    cx: &ParsingContext,
129    fields: &'a Punctuated<syn::Field, syn::Token![,]>,
130) -> Option<Vec<Field<'a>>> {
131    let result: Vec<Field<'_>> = fields
132        .iter()
133        .enumerate()
134        .flat_map(|(i, field)| {
135            Some(Field {
136                member: match &field.ident {
137                    Some(ident) => syn::Member::Named(ident.clone()),
138                    None => syn::Member::Unnamed(i.into()),
139                },
140                attrs: attr::Field::from_ast(cx, i, field)?,
141                ty: &field.ty,
142                original: field,
143            })
144        })
145        .collect();
146
147    (result.len() == fields.len()).then_some(result)
148}
