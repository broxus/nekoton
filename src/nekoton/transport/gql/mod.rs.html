37    async fn fetch<T>(&self, params: T::Variables) -> Result<T::ResponseData>
38    where
39        T: GqlQuery,
40    {
41        let request_body = serde_json::to_string(&T::build_query(&params)).trust_me();
42        let response = self
43            .connection
44            .post(GqlRequest {
45                data: request_body,
46                long_query: T::LONG_QUERY,
47            })
48            .await
49            .map_err(api_failure)?;
50
51        #[derive(Deserialize)]
52        pub struct Response<T> {
53            pub data: Option<T>,
54        }
55
56        match serde_json::from_str::<Response<T::ResponseData>>(&response) {
57            Ok(response) => response.data.ok_or_else(|| invalid_response().into()),
58            Err(e) => Err(api_failure(format!(
59                "Failed parsing api response: {e}. Response data: {response}"
60            ))
61            .into()),
62        }
63    }

65    pub async fn get_latest_block(&self, addr: &MsgAddressInt) -> Result<LatestBlock> {
66        let workchain_id = addr.get_workchain_id();
67
68        let block = self
69            .fetch::<QueryLatestMasterchainBlock>(())
70            .await?
71            .blocks
72            .into_iter()
73            .next();
74
75        match block {
76            Some(block) => {
77                // Handle simple case when searched account is in masterchain
78                if workchain_id == -1 {
79                    return Ok(LatestBlock {
80                        id: block.id,
81                        end_lt: parse_lt(&block.end_lt)?,
82                        gen_utime: block.gen_utime as u32,
83                    });
84                }
85
86                // Find matching shard
87                for item in block.master.shard_hashes {
88                    if check_shard_match(item.workchain_id, &item.shard, addr)? {
89                        return Ok(LatestBlock {
90                            id: item.descr.root_hash,
91                            end_lt: parse_lt(&item.descr.end_lt)?,
92                            gen_utime: item.descr.gen_utime as u32,
93                        });
94                    }
95                }
96
97                Err(no_blocks_found().into())
98            }
99            // Node SE case (without masterchain and sharding)
100            None => {
101                let blocks = self
102                    .fetch::<QueryNodeSeConditions>(query_node_se_conditions::Variables {
103                        workchain: workchain_id,
104                    })
105                    .await?
106                    .blocks;
107                let block = blocks.into_iter().next().ok_or_else(no_blocks_found)?;
108
109                // If workchain is sharded then it is not Node SE and missing masterchain blocks is error
110                if block.after_merge || block.shard != "8000000000000000" {
111                    return Err(no_blocks_found().into());
112                }
113
114                let blocks = self
115                    .fetch::<QueryNodeSeLatestBlock>(query_node_se_latest_block::Variables {
116                        workchain: workchain_id,
117                    })
118                    .await?
119                    .blocks;
120                let block = blocks.into_iter().next().ok_or_else(no_blocks_found)?;
121
122                Ok(LatestBlock {
123                    id: block.id,
124                    end_lt: parse_lt(&block.end_lt)?,
125                    gen_utime: block.gen_utime as u32,
126                })
127            }
128        }
129    }

131    pub async fn get_block(&self, id: &str) -> Result<ton_block::Block> {
132        let blocks = self

142    pub async fn wait_for_next_block(
143        &self,
144        current: &str,
145        addr: &MsgAddressInt,
146        timeout: Duration,
147    ) -> Result<String> {
148        let timeout_ms = timeout.as_secs_f64() * 1000.0;
149
150        let blocks = self
151            .fetch::<QueryNextBlock>(query_next_block::Variables {
152                id: current.to_owned(),
153                timeout: timeout_ms,
154            })
155            .await?
156            .blocks;
157        let block = blocks.into_iter().next().ok_or_else(no_blocks_found)?;
158
159        let block_id =
160            if block.after_split && !check_shard_match(block.workchain_id, &block.shard, addr)? {
161                let blocks = self
162                    .fetch::<QueryBlockAfterSplit>(query_block_after_split::Variables {
163                        block_id: block.id,
164                        prev_id: current.to_owned(),
165                        timeout: timeout_ms,
166                    })
167                    .await?
168                    .blocks;
169                blocks.into_iter().next().ok_or_else(no_blocks_found)?.id
170            } else {
171                block.id
172            };
173
174        Ok(block_id)
175    }

177    async fn fetch_config(&self) -> Result<ConfigResponse> {
178        let block = self.get_latest_key_block().await?;
179        let seqno = block.info.read_struct()?.seq_no();
180        let extra = block.read_extra()?;
181        let master = extra.read_custom()?.context("invalid key block")?;
182        let config = master.config().context("invalid key block")?.clone();
183
184        Ok(ConfigResponse {
185            global_id: block.global_id,
186            seqno,
187            config,
188        })
189    }

203    async fn send_message(&self, message: &Message) -> Result<()> {
204        let cell = message
205            .write_to_new_cell()
206            .and_then(ton_types::BuilderData::into_cell)
207            .map_err(|_| NodeClientError::FailedToSerialize)?;
208
209        let boc = base64::encode(
210            ton_types::serialize_toc(&cell).map_err(|_| NodeClientError::FailedToSerialize)?,
211        );
212        let id = base64::encode(cell.repr_hash());
213
214        let _ = self
215            .fetch::<MutationSendMessage>(mutation_send_message::Variables { id, boc })
216            .await?;
217
218        Ok(())
219    }

221    async fn get_contract_state(&self, address: &MsgAddressInt) -> Result<RawContractState> {
222        let account_state = match self
223            .fetch::<QueryAccountState>(query_account_state::Variables {
224                address: address.to_string(),
225            })
226            .await?
227            .accounts
228            .into_iter()
229            .next()
230            .and_then(|state| state.boc)
231        {
232            Some(boc) => boc,
233            None => {
234                return Ok(RawContractState::NotExists {
235                    timings: GenTimings::Unknown,
236                })
237            }
238        };
239
240        match Account::construct_from_base64(&account_state) {
241            Ok(Account::Account(account)) => {
242                let last_transaction_id = LastTransactionId::Inexact {
243                    latest_lt: account.storage.last_trans_lt,
244                };
245
246                Ok(RawContractState::Exists(ExistingContract {
247                    account,
248                    timings: GenTimings::Unknown,
249                    last_transaction_id,
250                }))
251            }
252            Ok(_) => Ok(RawContractState::NotExists {
253                timings: GenTimings::Unknown,
254            }),
255            Err(_) => Err(NodeClientError::InvalidAccountState.into()),
256        }
257    }

291    async fn get_transactions(
292        &self,
293        address: &MsgAddressInt,
294        from_lt: u64,
295        count: u8,
296    ) -> Result<Vec<RawTransaction>> {
297        self.fetch::<QueryAccountTransactions>(query_account_transactions::Variables {
298            address: address.to_string(),
299            last_transaction_lt: from_lt.to_string(),
300            limit: count,
301        })
302        .await?
303        .transactions
304        .into_iter()
305        .map(|transaction| {
306            let bytes = base64::decode(transaction.boc)?;
307            let cell = ton_types::deserialize_tree_of_cells(&mut bytes.as_slice())
308                .map_err(|_| NodeClientError::InvalidTransaction)?;
309            let hash = cell.repr_hash();
310            Ok(RawTransaction {
311                hash,
312                data: ton_block::Transaction::construct_from_cell(cell)
313                    .map_err(|_| NodeClientError::InvalidTransaction)?,
314            })
315        })
316        .collect()
317    }

402fn check_shard_match(workchain_id: i32, shard: &str, addr: &MsgAddressInt) -> Result<bool> {
403    let shard = u64::from_str_radix(shard, 16)?;
404
405    let ident =
406        ton_block::ShardIdent::with_tagged_prefix(workchain_id, shard).map_err(api_failure)?;
407
408    let prefix = ton_block::AccountIdPrefixFull::prefix(addr).map_err(api_failure)?;
409    Ok(ident.contains_full_prefix(&prefix))
410}

412fn api_failure<T>(e: T) -> NodeClientError
413where
414    T: std::fmt::Display,
415{
416    NodeClientError::ApiFailure {
417        reason: e.to_string(),
418    }
419}

421fn parse_lt(lt: &str) -> Result<u64, std::num::ParseIntError> {
422    match lt.strip_prefix("0x") {
423        Some(lt) => u64::from_str_radix(lt, 16),
424        None => u64::from_str(lt),
425    }
