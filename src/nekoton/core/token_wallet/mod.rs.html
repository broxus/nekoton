32    pub async fn subscribe(
33        clock: Arc<dyn Clock>,
34        transport: Arc<dyn Transport>,
35        owner: MsgAddressInt,
36        root_token_contract: MsgAddressInt,
37        handler: Arc<dyn TokenWalletSubscriptionHandler>,
38        preload_transactions: bool,
39    ) -> Result<TokenWallet> {
40        let state = match transport.get_contract_state(&root_token_contract).await? {
41            RawContractState::Exists(state) => state,
42            RawContractState::NotExists { .. } => {
43                return Err(TokenWalletError::InvalidRootTokenContract.into())
44            }
45        };
46        let state = RootTokenContractState(state.as_context(clock.as_ref()));
47        let RootTokenContractDetails {
48            symbol: name,
49            decimals,
50            version,
51            name: full_name,
52            ..
53        } = state.guess_details()?;
54
55        let address = state.get_wallet_address(version, &owner)?;
56        let mut balance = Default::default();
57
58        let contract_subscription = {
59            let handler = handler.as_ref();
60
61            // NOTE: create handler beforehead to prevent lifetime issues
62            let mut on_transactions_found = match preload_transactions {
63                true => Some(make_transactions_handler(handler, version)),
64                false => None,
65            };
66
67            // Manual map is used here due to unsoundness
68            // See issue: https://github.com/rust-lang/rust/issues/84305
69            #[allow(trivial_casts)]
70            #[allow(clippy::manual_map)]
71            let on_transactions_found = match &mut on_transactions_found {
72                Some(handler) => Some(handler as _),
73                None => None,
74            };
75
76            ContractSubscription::subscribe(
77                clock.clone(),
78                transport,
79                address,
80                &mut make_contract_state_handler(clock.clone(), version, &mut balance),
81                on_transactions_found,
82            )
83            .await?
84        };
85
86        handler.on_balance_changed(balance.clone());
87
88        let symbol = Symbol {
89            name,
90            full_name,
91            decimals,
92            root_token_contract,
93        };
94
95        Ok(Self {
96            clock,
97            contract_subscription,
98            handler,
99            owner,
100            symbol,
101            version,
102            balance,
103        })
104    }

134    pub async fn estimate_min_attached_amount(
135        &self,
136        destination: TransferRecipient,
137        tokens: BigUint,
138        notify_receiver: bool,
139        payload: ton_types::Cell,
140    ) -> Result<u64> {
141        const FEE_MULTIPLIER: u128 = 2;
142
143        // Prepare internal message
144        let internal_message = self
145            .prepare_transfer(destination, tokens, notify_receiver, payload, 0)
146            .await?;
147
148        let mut message = ton_block::Message::with_int_header(ton_block::InternalMessageHeader {
149            src: ton_block::MsgAddressIntOrNone::Some(
150                internal_message
151                    .source
152                    .unwrap_or_else(|| self.owner.clone()),
153            ),
154            dst: internal_message.destination,
155            ..Default::default()
156        });
157
158        message.set_body(internal_message.body.clone());
159
160        // Prepare executor
161        let transport = self.contract_subscription.transport().clone();
162        let config = transport
163            .get_blockchain_config(self.clock.as_ref(), true)
164            .await?;
165
166        let mut tree = TransactionsTreeStream::new(message, config, transport, self.clock.clone());
167        tree.unlimited_account_balance();
168        tree.unlimited_message_balance();
169
170        type Err = fn(Option<i32>) -> TokenWalletError;
171        let check_exit_code = |tx: &ton_block::Transaction, err: Err| -> Result<()> {
172            let descr = tx.read_description()?;
173            if descr.is_aborted() {
174                let exit_code = match descr {
175                    ton_block::TransactionDescr::Ordinary(descr) => match descr.compute_ph {
176                        ton_block::TrComputePhase::Vm(phase) => Some(phase.exit_code),
177                        ton_block::TrComputePhase::Skipped(_) => None,
178                    },
179                    _ => None,
180                };
181                Err(err(exit_code).into())
182            } else {
183                Ok(())
184            }
185        };
186
187        let mut attached_amount: u128 = 0;
188
189        // Simulate source transaction
190        let source_tx = tree.next().await?.ok_or(TokenWalletError::NoSourceTx)?;
191        check_exit_code(&source_tx, TokenWalletError::SourceTxFailed)?;
192        attached_amount += source_tx.total_fees.grams.as_u128();
193
194        if source_tx.outmsg_cnt == 0 {
195            return Err(TokenWalletError::NoDestTx.into());
196        }
197
198        if let Some(message) = tree.peek() {
199            if message.state_init().is_some() && message.src_ref() == Some(self.address()) {
200                // Simulate first deploy transaction
201                // NOTE: we don't need to count attached amount here because of separate `initial_balance`
202                let _ = tree.next().await?.ok_or(TokenWalletError::NoDestTx)?;
203                //also we ignore non zero exit code for deploy transactions
204            }
205        }
206
207        tree.retain_message_queue(|message| {
208            message.state_init().is_none() && message.src_ref() == Some(self.address())
209        });
210
211        if tree.message_queue().len() != 1 {
212            return Err(TokenWalletError::NoDestTx.into());
213        }
214
215        // Simulate destination transaction
216        let dest_tx = tree.next().await?.ok_or(TokenWalletError::NoDestTx)?;
217        check_exit_code(&dest_tx, TokenWalletError::DestinationTxFailed)?;
218        attached_amount += dest_tx.total_fees.grams.as_u128();
219
220        Ok((attached_amount * FEE_MULTIPLIER) as u64)
221    }

223    pub async fn prepare_transfer(
224        &self,
225        destination: TransferRecipient,
226        tokens: BigUint,
227        notify_receiver: bool,
228        payload: ton_types::Cell,
229        mut attached_amount: u128,
230    ) -> Result<InternalMessage> {
231        fn stub_balance(address: &MsgAddressInt) -> u128 {
232            if address.is_masterchain() {
233                1000
234            } else {
235                1
236            }
237        }
238
239        let initial_balance = match &destination {
240            TransferRecipient::OwnerWallet(address) => {
241                let transport = self.contract_subscription.transport();
242                let config = transport
243                    .get_blockchain_config(self.clock.as_ref(), true)
244                    .await?;
245
246                let initial_balance = match calculate_initial_balance(&config, address) {
247                    Ok(value) => value + stub_balance(address),
248                    Err(_) => stub_balance(address),
249                };
250                attached_amount += initial_balance;
251                initial_balance
252            }
253            TransferRecipient::TokenWallet(address) => stub_balance(address),
254        };
255
256        let (function, input) = match self.version {
257            TokenWalletVersion::OldTip3v4 => {
258                use old_tip3::token_wallet_contract;
259                match destination {
260                    TransferRecipient::TokenWallet(token_wallet) => {
261                        MessageBuilder::new(token_wallet_contract::transfer())
262                            .arg(token_wallet) // to
263                            .arg(BigUint128(tokens)) // tokens
264                    }
265                    TransferRecipient::OwnerWallet(owner_wallet) => {
266                        MessageBuilder::new(token_wallet_contract::transfer_to_recipient())
267                            .arg(BigUint256(Default::default())) // recipient_public_key
268                            .arg(owner_wallet) // recipient_address
269                            .arg(BigUint128(tokens)) // tokens
270                            .arg(BigUint128(initial_balance.into())) // deploy_grams
271                    }
272                }
273                .arg(BigUint128(Default::default())) // grams / transfer_grams
274                .arg(&self.owner) // send_gas_to
275                .arg(notify_receiver) // notify_receiver
276                .arg(payload) // payload
277                .build()
278            }
279            TokenWalletVersion::Tip3 => {
280                use tip3_1::token_wallet_contract;
281                match destination {
282                    TransferRecipient::TokenWallet(token_wallet) => {
283                        MessageBuilder::new(token_wallet_contract::transfer_to_wallet())
284                            .arg(BigUint128(tokens)) // amount
285                            .arg(token_wallet) // recipient token wallet
286                    }
287                    TransferRecipient::OwnerWallet(owner_wallet) => {
288                        MessageBuilder::new(token_wallet_contract::transfer())
289                            .arg(BigUint128(tokens)) // amount
290                            .arg(owner_wallet) // recipient
291                            .arg(BigUint128(initial_balance.into())) // deployWalletValue
292                    }
293                }
294                .arg(&self.owner) // remainingGasTo
295                .arg(notify_receiver) // notify
296                .arg(payload) // payload
297                .build()
298            }
299        };
300
301        let body = function
302            .encode_internal_input(&input)
303            .and_then(ton_types::SliceData::load_builder)?;
304
305        Ok(InternalMessage {
306            source: Some(self.owner.clone()),
307            destination: self.address().clone(),
308            amount: attached_amount,
309            bounce: true,
310            body,
311        })
312    }

315        let mut balance = self.balance.clone();

322                &mut |_, _| {},
323                &mut |_| {},

329            handler.on_balance_changed(self.balance.clone());

335    pub async fn handle_block(&mut self, block: &ton_block::Block) -> Result<()> {
336        let version = self.version;
337        let mut balance: BigInt = self.balance.clone().into();
338
339        let handler = self.handler.as_ref();
340        self.contract_subscription.handle_block(
341            block,
342            &mut |transactions, batch_info| {
343                let transactions = transactions
344                    .into_iter()
345                    .filter_map(|transaction| {
346                        let description = match transaction.data.description.read_struct().ok()? {
347                            ton_block::TransactionDescr::Ordinary(description) => description,
348                            _ => return None,
349                        };
350
351                        let data =
352                            parse_token_transaction(&transaction.data, &description, version);
353
354                        if let Some(data) = &data {
355                            match data {
356                                TokenWalletTransaction::IncomingTransfer(
357                                    TokenIncomingTransfer { tokens, .. },
358                                )
359                                | TokenWalletTransaction::Accept(tokens)
360                                | TokenWalletTransaction::SwapBackBounced(tokens)
361                                | TokenWalletTransaction::TransferBounced(tokens) => {
362                                    balance += tokens.clone().to_bigint().trust_me();
363                                }
364                                TokenWalletTransaction::OutgoingTransfer(
365                                    TokenOutgoingTransfer { tokens, .. },
366                                )
367                                | TokenWalletTransaction::SwapBack(TokenSwapBack {
368                                    tokens, ..
369                                }) => {
370                                    balance -= tokens.clone().to_bigint().trust_me();
371                                }
372                            }
373                        }
374
375                        let transaction =
376                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
377
378                        Some(TransactionWithData { transaction, data })
379                    })
380                    .collect();
381
382                handler.on_transactions_found(transactions, batch_info)
383            },
384            &mut |_, _| {},
385            &mut |_| {},
386        )?;
387
388        let balance = balance.to_biguint().unwrap_or_default();
389        if balance != self.balance {
390            self.balance = balance;
391            handler.on_balance_changed(self.balance.clone());
392        }
393
394        Ok(())
395    }

510fn calculate_initial_balance(config: &BlockchainConfig, address: &MsgAddressInt) -> Result<u128> {
511    let gas_config = config.get_gas_config(address.is_masterchain());
512
513    let prices = config.raw_config().storage_prices()?;
514    let mut most_recent_bit_price = 0;
515    let mut most_recent_mc_bit_price = 0;
516    let mut most_recent_time: i32 = -1;
517
518    prices.map.iterate(|price| {
519        if most_recent_time < price.utime_since as i32 {
520            most_recent_time = price.utime_since as i32;
521            most_recent_bit_price = price.bit_price_ps;
522            most_recent_mc_bit_price = price.mc_bit_price_ps;
523        }
524        Ok(true)
525    })?;
526
527    let storage_fees = if address.is_masterchain() {
528        most_recent_mc_bit_price
529    } else {
530        most_recent_bit_price
531    };
532
533    let fee_base = if storage_fees > 1 {
534        600_000_000u64
535    } else {
536        100_000_000u64
537    };
538
539    let fees = 30000u64.saturating_mul(gas_config.gas_price.shr(16)) + fee_base;
540
541    Ok(fees as u128)
542}

544fn make_transactions_handler(
545    handler: &'_ dyn TokenWalletSubscriptionHandler,
546    version: TokenWalletVersion,
547) -> impl FnMut(Vec<RawTransaction>, TransactionsBatchInfo) + '_ {
548    move |transactions, batch_info| {
549        let transactions = transactions
550            .into_iter()
551            .filter_map(
552                |transaction| match transaction.data.description.read_struct().ok()? {
553                    ton_block::TransactionDescr::Ordinary(description) => {
554                        let data =
555                            parse_token_transaction(&transaction.data, &description, version);
556
557                        let transaction =
558                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
559
560                        Some(TransactionWithData { transaction, data })
561                    }
562                    _ => None,
563                },
564            )
565            .collect();
566
567        handler.on_transactions_found(transactions, batch_info)
568    }
569}
