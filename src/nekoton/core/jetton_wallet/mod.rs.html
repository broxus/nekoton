33    pub async fn subscribe(
34        clock: Arc<dyn Clock>,
35        transport: Arc<dyn Transport>,
36        owner: MsgAddressInt,
37        root_token_contract: MsgAddressInt,
38        handler: Arc<dyn JettonWalletSubscriptionHandler>,
39        preload_transactions: bool,
40    ) -> Result<JettonWallet> {
41        let state = match transport.get_contract_state(&root_token_contract).await? {
42            RawContractState::Exists(state) => state,
43            RawContractState::NotExists { .. } => {
44                return Err(JettonWalletError::InvalidRootTokenContract.into())
45            }
46        };
47        let state = nekoton_contracts::jetton::RootTokenContract(state.as_context(clock.as_ref()));
48
49        let address = state.get_wallet_address(&owner)?;
50        let mut balance = Default::default();
51
52        let contract_subscription = {
53            let handler = handler.as_ref();
54            // NOTE: create handler beforehead to prevent lifetime issues
55            let mut on_transactions_found = match preload_transactions {
56                true => Some(make_transactions_handler(handler)),
57                false => None,
58            };
59
60            // Manual map is used here due to unsoundness
61            // See issue: https://github.com/rust-lang/rust/issues/84305
62            #[allow(trivial_casts)]
63            #[allow(clippy::manual_map)]
64            let on_transactions_found = match &mut on_transactions_found {
65                Some(handler) => Some(handler as _),
66                None => None,
67            };
68
69            ContractSubscription::subscribe(
70                clock.clone(),
71                transport,
72                address,
73                &mut make_contract_state_handler(clock.clone(), &mut balance),
74                on_transactions_found,
75            )
76            .await?
77        };
78
79        handler.on_balance_changed(balance.clone());
80
81        Ok(Self {
82            clock,
83            contract_subscription,
84            handler,
85            owner,
86            balance,
87            root: root_token_contract,
88        })
89    }

134    pub fn prepare_transfer(
135        &self,
136        amount: BigUint,
137        destination: MsgAddressInt,
138        remaining_gas_to: MsgAddressInt,
139        custom_payload: Option<ton_types::Cell>,
140        callback_value: BigUint,
141        callback_payload: Option<ton_types::Cell>,
142        attached_amount: u128,
143    ) -> Result<InternalMessage> {
144        let mut builder = BuilderData::new();
145
146        // Opcode
147        builder.append_u32(JETTON_TRANSFER_OPCODE)?;
148
149        // Query id
150        builder.append_u64(self.clock.now_ms_u64())?;
151
152        // Amount
153        let grams =
154            ton_block::Grams::new(amount.to_u128().ok_or(JettonWalletError::TryFromGrams)?)?;
155        grams.write_to(&mut builder)?;
156
157        // Recipient
158        destination.write_to(&mut builder)?;
159
160        // Response destination
161        remaining_gas_to.write_to(&mut builder)?;
162
163        // Optional(TvmCell)
164        match custom_payload {
165            Some(payload) => {
166                builder.append_bit_one()?;
167                builder.checked_append_reference(payload)?;
168            }
169            None => {
170                builder.append_bit_zero()?;
171            }
172        }
173
174        // Callback value
175        let grams = ton_block::Grams::new(
176            callback_value
177                .to_u128()
178                .ok_or(JettonWalletError::TryFromGrams)?,
179        )?;
180        grams.write_to(&mut builder)?;
181
182        let callback_payload = callback_payload.unwrap_or_default();
183        if callback_payload.bit_length() < builder.bits_free() {
184            builder.append_bit_zero()?;
185
186            let callback_builder = BuilderData::from_cell(&callback_payload);
187            builder.append_builder(&callback_builder)?;
188        } else {
189            builder.append_bit_one()?;
190            builder.checked_append_reference(callback_payload)?;
191        }
192
193        let body = builder.into_cell().map(SliceData::load_cell)??;
194
195        Ok(InternalMessage {
196            source: Some(self.owner.clone()),
197            destination: self.address().clone(),
198            amount: attached_amount,
199            bounce: true,
200            body,
201        })
202    }

204    pub async fn refresh(&mut self) -> Result<()> {
205        let mut balance = self.balance.clone();
206
207        let handler = self.handler.as_ref();
208        self.contract_subscription
209            .refresh(
210                &mut make_contract_state_handler(self.clock.clone(), &mut balance),
211                &mut make_transactions_handler(handler),
212                &mut |_, _| {},
213                &mut |_| {},
214            )
215            .await?;
216
217        if balance != self.balance {
218            self.balance = balance;
219            handler.on_balance_changed(self.balance.clone());
220        }
221
222        Ok(())
223    }

225    pub async fn handle_block(&mut self, block: &ton_block::Block) -> Result<()> {
226        let mut balance: BigInt = self.balance.clone().into();
227
228        let handler = self.handler.as_ref();
229        self.contract_subscription.handle_block(
230            block,
231            &mut |transactions, batch_info| {
232                let transactions = transactions
233                    .into_iter()
234                    .filter_map(|transaction| {
235                        let description = match transaction.data.description.read_struct().ok()? {
236                            ton_block::TransactionDescr::Ordinary(description) => description,
237                            _ => return None,
238                        };
239
240                        let data = parse_jetton_transaction(&transaction.data, &description);
241
242                        if let Some(data) = &data {
243                            match data {
244                                JettonWalletTransaction::Transfer(transfer) => {
245                                    balance -= transfer.tokens.clone().to_bigint().trust_me();
246                                }
247                                JettonWalletTransaction::InternalTransfer(transfer) => {
248                                    balance += transfer.tokens.clone().to_bigint().trust_me();
249                                }
250                                JettonWalletTransaction::BurnNotification(transfer) => {
251                                    balance -= transfer.tokens.clone().to_bigint().trust_me();
252                                }
253                            }
254                        }
255
256                        let transaction =
257                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
258
259                        Some(TransactionWithData { transaction, data })
260                    })
261                    .collect();
262
263                handler.on_transactions_found(transactions, batch_info)
264            },
265            &mut |_, _| {},
266            &mut |_| {},
267        )?;
268
269        let balance = balance.to_biguint().unwrap_or_default();
270        if balance != self.balance {
271            self.balance = balance;
272            handler.on_balance_changed(self.balance.clone());
273        }
274
275        Ok(())
276    }

425fn make_contract_state_handler(
426    clock: Arc<dyn Clock>,
427    balance: &'_ mut BigUint,
428) -> impl FnMut(&RawContractState) + '_ {
429    move |contract_state| {
430        if let RawContractState::Exists(state) = contract_state {
431            if let Ok(new_balance) =
432                nekoton_contracts::jetton::TokenWalletContract(state.as_context(clock.as_ref()))
433                    .balance()
434            {
435                *balance = new_balance
436            }
437        }
438    }

441fn make_transactions_handler(
442    handler: &'_ dyn JettonWalletSubscriptionHandler,
443) -> impl FnMut(Vec<RawTransaction>, TransactionsBatchInfo) + '_ {
444    move |transactions, batch_info| {
445        let transactions = transactions
446            .into_iter()
447            .filter_map(
448                |transaction| match transaction.data.description.read_struct().ok()? {
449                    ton_block::TransactionDescr::Ordinary(description) => {
450                        let data = parse_jetton_transaction(&transaction.data, &description);
451
452                        let transaction =
453                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
454
455                        Some(TransactionWithData { transaction, data })
456                    }
457                    _ => None,
458                },
459            )
460            .collect();
461
462        handler.on_transactions_found(transactions, batch_info)
463    }
