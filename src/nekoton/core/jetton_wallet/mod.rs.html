32    pub async fn subscribe(
33        clock: Arc<dyn Clock>,
34        transport: Arc<dyn Transport>,
35        owner: MsgAddressInt,
36        root_token_contract: MsgAddressInt,
37        handler: Arc<dyn JettonWalletSubscriptionHandler>,
38        preload_transactions: bool,
39    ) -> Result<JettonWallet> {
40        let state = match transport.get_contract_state(&root_token_contract).await? {
41            RawContractState::Exists(state) => state,
42            RawContractState::NotExists { .. } => {
43                return Err(JettonWalletError::InvalidRootTokenContract.into())
44            }
45        };
46        let state = nekoton_contracts::jetton::RootTokenContract(state.as_context(clock.as_ref()));
47
48        let address = state.get_wallet_address(&owner)?;
49        let mut balance = Default::default();
50
51        let contract_subscription = {
52            let handler = handler.as_ref();
53            // NOTE: create handler beforehead to prevent lifetime issues
54            let mut on_transactions_found = match preload_transactions {
55                true => Some(make_transactions_handler(handler)),
56                false => None,
57            };
58
59            // Manual map is used here due to unsoundness
60            // See issue: https://github.com/rust-lang/rust/issues/84305
61            #[allow(trivial_casts)]
62            #[allow(clippy::manual_map)]
63            let on_transactions_found = match &mut on_transactions_found {
64                Some(handler) => Some(handler as _),
65                None => None,
66            };
67
68            ContractSubscription::subscribe(
69                clock.clone(),
70                transport,
71                address,
72                &mut make_contract_state_handler(clock.clone(), &mut balance),
73                on_transactions_found,
74            )
75            .await?
76        };
77
78        handler.on_balance_changed(balance.clone());
79
80        Ok(Self {
81            clock,
82            contract_subscription,
83            handler,
84            owner,
85            balance,
86            root: root_token_contract,
87        })
88    }

133    pub fn prepare_transfer(
134        &self,
135        amount: BigUint,
136        destination: MsgAddressInt,
137        remaining_gas_to: MsgAddressInt,
138        custom_payload: Option<ton_types::Cell>,
139        callback_value: BigUint,
140        callback_payload: Option<ton_types::Cell>,
141        attached_amount: u128,
142    ) -> Result<InternalMessage> {
143        let mut builder = BuilderData::new();
144
145        // Opcode
146        builder.append_u32(JETTON_TRANSFER_OPCODE)?;
147
148        // Query id
149        builder.append_u64(self.clock.now_ms_u64())?;
150
151        // Amount
152        let grams =
153            ton_block::Grams::new(amount.to_u128().ok_or(JettonWalletError::TryFromGrams)?)?;
154        grams.write_to(&mut builder)?;
155
156        // Recipient
157        destination.write_to(&mut builder)?;
158
159        // Response destination
160        remaining_gas_to.write_to(&mut builder)?;
161
162        // Optional(TvmCell)
163        match custom_payload {
164            Some(payload) => {
165                builder.append_bit_one()?;
166                builder.checked_append_reference(payload)?;
167            }
168            None => {
169                builder.append_bit_zero()?;
170            }
171        }
172
173        // Callback value
174        let grams = ton_block::Grams::new(
175            callback_value
176                .to_u128()
177                .ok_or(JettonWalletError::TryFromGrams)?,
178        )?;
179        grams.write_to(&mut builder)?;
180
181        let callback_payload = callback_payload.unwrap_or_default();
182        if callback_payload.bit_length() < builder.bits_free() {
183            builder.append_bit_zero()?;
184
185            let callback_builder = BuilderData::from_cell(&callback_payload);
186            builder.append_builder(&callback_builder)?;
187        } else {
188            builder.append_bit_one()?;
189            builder.checked_append_reference(callback_payload)?;
190        }
191
192        let body = builder.into_cell().map(SliceData::load_cell)??;
193
194        Ok(InternalMessage {
195            source: Some(self.owner.clone()),
196            destination: self.address().clone(),
197            amount: attached_amount,
198            bounce: true,
199            body,
200        })
201    }

203    pub async fn refresh(&mut self) -> Result<()> {
204        let mut balance = self.balance.clone();
205
206        let handler = self.handler.as_ref();
207        self.contract_subscription
208            .refresh(
209                &mut make_contract_state_handler(self.clock.clone(), &mut balance),
210                &mut make_transactions_handler(handler),
211                &mut |_, _| {},
212                &mut |_| {},
213            )
214            .await?;
215
216        if balance != self.balance {
217            self.balance = balance;
218            handler.on_balance_changed(self.balance.clone());
219        }
220
221        Ok(())
222    }

224    pub async fn handle_block(&mut self, block: &ton_block::Block) -> Result<()> {
225        let mut balance: BigInt = self.balance.clone().into();
226
227        let handler = self.handler.as_ref();
228        self.contract_subscription.handle_block(
229            block,
230            &mut |transactions, batch_info| {
231                let transactions = transactions
232                    .into_iter()
233                    .filter_map(|transaction| {
234                        let description = match transaction.data.description.read_struct().ok()? {
235                            ton_block::TransactionDescr::Ordinary(description) => description,
236                            _ => return None,
237                        };
238
239                        let data = parse_jetton_transaction(&transaction.data, &description);
240
241                        if let Some(data) = &data {
242                            match data {
243                                JettonWalletTransaction::Transfer(transfer) => {
244                                    balance -= transfer.tokens.clone().to_bigint().trust_me();
245                                }
246                                JettonWalletTransaction::InternalTransfer(transfer) => {
247                                    balance += transfer.tokens.clone().to_bigint().trust_me();
248                                }
249                            }
250                        }
251
252                        let transaction =
253                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
254
255                        Some(TransactionWithData { transaction, data })
256                    })
257                    .collect();
258
259                handler.on_transactions_found(transactions, batch_info)
260            },
261            &mut |_, _| {},
262            &mut |_| {},
263        )?;
264
265        let balance = balance.to_biguint().unwrap_or_default();
266        if balance != self.balance {
267            self.balance = balance;
268            handler.on_balance_changed(self.balance.clone());
269        }
270
271        Ok(())
272    }

421fn make_contract_state_handler(
422    clock: Arc<dyn Clock>,
423    balance: &'_ mut BigUint,
424) -> impl FnMut(&RawContractState) + '_ {
425    move |contract_state| {
426        if let RawContractState::Exists(state) = contract_state {
427            if let Ok(new_balance) =
428                nekoton_contracts::jetton::TokenWalletContract(state.as_context(clock.as_ref()))
429                    .balance()
430            {
431                *balance = new_balance
432            }
433        }
434    }

437fn make_transactions_handler(
438    handler: &'_ dyn JettonWalletSubscriptionHandler,
439) -> impl FnMut(Vec<RawTransaction>, TransactionsBatchInfo) + '_ {
440    move |transactions, batch_info| {
441        let transactions = transactions
442            .into_iter()
443            .filter_map(
444                |transaction| match transaction.data.description.read_struct().ok()? {
445                    ton_block::TransactionDescr::Ordinary(description) => {
446                        let data = parse_jetton_transaction(&transaction.data, &description);
447
448                        let transaction =
449                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
450
451                        Some(TransactionWithData { transaction, data })
452                    }
453                    _ => None,
454                },
455            )
456            .collect();
457
458        handler.on_transactions_found(transactions, batch_info)
459    }
