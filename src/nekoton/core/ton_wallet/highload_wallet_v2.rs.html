14pub fn prepare_deploy(
15    clock: &dyn Clock,
16    public_key: &PublicKey,
17    workchain: i8,
18    expiration: Expiration,
19) -> Result<Box<dyn UnsignedMessage>> {
20    let init_data = InitData::from_key(public_key).with_wallet_id(WALLET_ID);
21    let dst = compute_contract_address(public_key, workchain);
22    let mut message =
23        ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
24            dst,
25            ..Default::default()
26        });
27
28    message.set_state_init(init_data.make_state_init()?);
29
30    let expire_at = ExpireAt::new(clock, expiration);
31    let (hash, payload) = init_data.make_deploy_payload(expire_at.timestamp)?;
32
33    Ok(Box::new(UnsignedHighloadWalletV2Message {
34        init_data,
35        gifts: Vec::new(),
36        payload,
37        message,
38        expire_at,
39        hash,
40    }))
41}

48pub fn prepare_transfer(
49    clock: &dyn Clock,
50    public_key: &PublicKey,
51    current_state: &ton_block::AccountStuff,
52    gifts: Vec<Gift>,
53    expiration: Expiration,
54) -> Result<TransferAction> {
55    if gifts.len() > DETAILS.max_messages {
56        return Err(HighloadWalletV2Error::TooManyGifts.into());
57    }
58
59    let (init_data, with_state_init) = match &current_state.storage.state {
60        ton_block::AccountState::AccountActive { state_init, .. } => match &state_init.data {
61            Some(data) => (InitData::try_from(data)?, false),
62            None => return Err(HighloadWalletV2Error::InvalidInitData.into()),
63        },
64        ton_block::AccountState::AccountFrozen { .. } => {
65            return Err(HighloadWalletV2Error::AccountIsFrozen.into())
66        }
67        ton_block::AccountState::AccountUninit => (
68            InitData::from_key(public_key).with_wallet_id(WALLET_ID),
69            true,
70        ),
71    };
72
73    if init_data.data.len()? >= 500_usize {
74        return Err(HighloadWalletV2Error::InitDataTooLarge.into());
75    }
76
77    let mut message =
78        ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
79            dst: current_state.addr.clone(),
80            ..Default::default()
81        });
82
83    if with_state_init {
84        message.set_state_init(init_data.make_state_init()?);
85    }
86
87    let expire_at = ExpireAt::new(clock, expiration);
88    let (hash, payload) = init_data.make_transfer_payload(gifts.clone(), expire_at.timestamp)?;
89
90    Ok(TransferAction::Sign(Box::new(
91        UnsignedHighloadWalletV2Message {
92            init_data,
93            gifts,
94            payload,
95            hash,
96            expire_at,
97            message,
98        },
99    )))
100}

102#[derive(Clone)]
103struct UnsignedHighloadWalletV2Message {
104    init_data: InitData,
105    gifts: Vec<Gift>,
106    payload: BuilderData,
107    hash: UInt256,
108    expire_at: ExpireAt,
109    message: ton_block::Message,
110}
111
112impl UnsignedMessage for UnsignedHighloadWalletV2Message {
113    fn refresh_timeout(&mut self, clock: &dyn Clock) {
114        if !self.expire_at.refresh(clock) {
115            return;
116        }
117
118        let expire_at = self.expire_at();
119
120        let (hash, payload) = if self.gifts.is_empty() {
121            self.init_data.make_deploy_payload(expire_at)
122        } else {
123            self.init_data
124                .make_transfer_payload(self.gifts.clone(), expire_at)
125        }
126        .trust_me();
127
128        self.hash = hash;
129        self.payload = payload;
130    }
131
132    fn expire_at(&self) -> u32 {
133        self.expire_at.timestamp
134    }
135
136    fn hash(&self) -> &[u8] {
137        self.hash.as_slice()
138    }
139
140    fn sign(&self, signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH]) -> Result<SignedMessage> {
141        let mut payload = self.payload.clone();
142        payload.prepend_raw(signature, signature.len() * 8)?;
143
144        let mut message = self.message.clone();
145        message.set_body(ton_types::SliceData::load_builder(payload)?);
146
147        Ok(SignedMessage {
148            message,
149            expire_at: self.expire_at(),
150        })
151    }
152
153    fn sign_with_pruned_payload(
154        &self,
155        signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH],
156        prune_after_depth: u16,
157    ) -> Result<SignedMessage> {
158        let mut payload = self.payload.clone();
159        payload.prepend_raw(signature, signature.len() * 8)?;
160        let body = payload.into_cell()?;
161
162        let mut message = self.message.clone();
163        message.set_body(prune_deep_cells(&body, prune_after_depth)?);
164
165        Ok(SignedMessage {
166            message,
167            expire_at: self.expire_at(),
168        })
169    }
170}

200/// `HighloadWalletV2` init data
201#[derive(Clone)]
202pub struct InitData {
203    pub wallet_id: u32,
204    pub last_cleaned: u64,
205    pub public_key: UInt256,
206    pub data: HashmapE,
207}
208
209impl InitData {
210    pub fn public_key(&self) -> &[u8; 32] {
211        self.public_key.as_slice()
212    }
213
214    pub fn from_key(key: &PublicKey) -> Self {
215        Self {
216            wallet_id: 0,
217            last_cleaned: 0,
218            public_key: key.as_bytes().into(),
219            data: HashmapE::default(),
220        }
221    }
222
223    pub fn with_wallet_id(mut self, id: u32) -> Self {
224        self.wallet_id = id;
225        self
226    }
227
228    pub fn compute_addr(&self, workchain_id: i8) -> Result<MsgAddressInt> {
229        let init_state = self.make_state_init()?.serialize()?;
230        let hash = init_state.repr_hash();
231        Ok(MsgAddressInt::AddrStd(MsgAddrStd {
232            anycast: None,
233            workchain_id,
234            address: hash.into(),
235        }))
236    }
237
238    pub fn make_state_init(&self) -> Result<ton_block::StateInit> {
239        Ok(ton_block::StateInit {
240            code: Some(nekoton_contracts::wallets::code::highload_wallet_v2()),
241            data: Some(self.serialize()?),
242            ..Default::default()
243        })
244    }
245
246    pub fn serialize(&self) -> Result<Cell> {
247        let mut data = BuilderData::new();
248        data.append_u32(self.wallet_id)?
249            .append_u64(self.last_cleaned)?
250            .append_raw(self.public_key.as_slice(), 256)?;
251        self.data.write_hashmap_data(&mut data)?;
252        data.into_cell()
253    }
254
255    pub fn make_deploy_payload(&self, expire_at: u32) -> Result<(UInt256, BuilderData)> {
256        let mut payload = BuilderData::new();
257        payload
258            .append_u32(self.wallet_id)?
259            .append_u32(expire_at)?
260            .append_u32(u32::MAX)?
261            .append_bit_zero()?;
262
263        let hash = payload.clone().into_cell()?.repr_hash();
264
265        Ok((hash, payload))
266    }
267
268    pub fn make_transfer_payload(
269        &self,
270        gifts: impl IntoIterator<Item = Gift>,
271        expire_at: u32,
272    ) -> Result<(UInt256, BuilderData)> {
273        // Prepare messages array
274        let mut messages = ton_types::HashmapE::with_bit_len(16);
275        for (i, gift) in gifts.into_iter().enumerate() {
276            let mut internal_message =
277                ton_block::Message::with_int_header(ton_block::InternalMessageHeader {
278                    ihr_disabled: true,
279                    bounce: gift.bounce,
280                    dst: gift.destination,
281                    value: ton_block::CurrencyCollection::from_grams(ton_block::Grams::new(
282                        gift.amount,
283                    )?),
284                    ..Default::default()
285                });
286
287            if let Some(body) = gift.body {
288                internal_message.set_body(body);
289            }
290
291            if let Some(state_init) = gift.state_init {
292                internal_message.set_state_init(state_init);
293            }
294
295            let mut item = BuilderData::new();
296            item.append_u8(gift.flags)?
297                .checked_append_reference(internal_message.serialize()?)?;
298
299            let key = (i as u16)
300                .serialize()
301                .and_then(SliceData::load_cell)
302                .trust_me();
303
304            messages.set_builder(key, &item)?;
305        }
306
307        let messages = messages.serialize()?;
308        let messages_hash = messages.repr_hash();
309
310        // Build payload
311        let mut payload = BuilderData::new();
312        payload
313            .append_u32(self.wallet_id)?
314            .append_u32(expire_at)?
315            .append_raw(&messages_hash.as_slice()[28..32], 32)?
316            .append_builder(&messages.into())?;
317
318        let hash = payload.clone().into_cell()?.repr_hash();
319
320        Ok((hash, payload))
321    }
322}

324impl TryFrom<&Cell> for InitData {
325    type Error = anyhow::Error;
326
327    fn try_from(data: &Cell) -> Result<Self, Self::Error> {
328        let mut cs = SliceData::load_cell_ref(data)?;
329
330        Ok(Self {
331            wallet_id: cs.get_next_u32()?,
332            last_cleaned: cs.get_next_u64()?,
333            public_key: UInt256::from_be_bytes(&cs.get_next_bytes(32)?),
334            data: {
335                let mut map = HashmapE::with_bit_len(64);
336                map.read_hashmap_data(&mut cs)?;
337                map
338            },
339        })
340    }
341}
