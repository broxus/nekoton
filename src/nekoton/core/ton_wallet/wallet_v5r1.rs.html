16pub fn prepare_deploy(
17    clock: &dyn Clock,
18    public_key: &PublicKey,
19    workchain: i8,
20    expiration: Expiration,
21) -> Result<Box<dyn UnsignedMessage>> {
22    let init_data = InitData::from_key(public_key)
23        .with_wallet_id(WALLET_ID)
24        .with_is_signature_allowed(true);
25    let dst = compute_contract_address(public_key, workchain);
26    let mut message =
27        ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
28            dst,
29            ..Default::default()
30        });
31
32    message.set_state_init(init_data.make_state_init()?);
33
34    let expire_at = ExpireAt::new(clock, expiration);
35    let (hash, payload) = init_data.make_transfer_payload(None, expire_at.timestamp)?;
36
37    Ok(Box::new(UnsignedWalletV5 {
38        init_data,
39        gifts: Vec::new(),
40        payload,
41        message,
42        expire_at,
43        hash,
44    }))
45}

54pub fn prepare_transfer(
55    clock: &dyn Clock,
56    public_key: &PublicKey,
57    current_state: &ton_block::AccountStuff,
58    seqno_offset: u32,
59    gifts: Vec<Gift>,
60    expiration: Expiration,
61) -> Result<TransferAction> {
62    if gifts.len() > MAX_MESSAGES {
63        return Err(WalletV5Error::TooManyGifts.into());
64    }
65
66    let (mut init_data, with_state_init) = match &current_state.storage.state {
67        ton_block::AccountState::AccountActive { state_init, .. } => match &state_init.data {
68            Some(data) => (InitData::try_from(data)?, false),
69            None => return Err(WalletV5Error::InvalidInitData.into()),
70        },
71        ton_block::AccountState::AccountFrozen { .. } => {
72            return Err(WalletV5Error::AccountIsFrozen.into())
73        }
74        ton_block::AccountState::AccountUninit => (
75            InitData::from_key(public_key)
76                .with_wallet_id(WALLET_ID)
77                .with_is_signature_allowed(true),
78            true,
79        ),
80    };
81
82    init_data.seqno += seqno_offset;
83
84    let mut message =
85        ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
86            dst: current_state.addr.clone(),
87            ..Default::default()
88        });
89
90    if with_state_init {
91        message.set_state_init(init_data.make_state_init()?);
92    }
93
94    let expire_at = ExpireAt::new(clock, expiration);
95    let (hash, payload) = init_data.make_transfer_payload(gifts.clone(), expire_at.timestamp)?;
96
97    Ok(TransferAction::Sign(Box::new(UnsignedWalletV5 {
98        init_data,
99        gifts,
100        payload,
101        hash,
102        expire_at,
103        message,
104    })))
105}

107#[derive(Clone)]
108struct UnsignedWalletV5 {
109    init_data: InitData,
110    gifts: Vec<Gift>,
111    payload: BuilderData,
112    hash: UInt256,
113    expire_at: ExpireAt,
114    message: ton_block::Message,
115}
116
117impl UnsignedMessage for UnsignedWalletV5 {
118    fn refresh_timeout(&mut self, clock: &dyn Clock) {
119        if !self.expire_at.refresh(clock) {
120            return;
121        }
122
123        let (hash, payload) = self
124            .init_data
125            .make_transfer_payload(self.gifts.clone(), self.expire_at())
126            .trust_me();
127        self.hash = hash;
128        self.payload = payload;
129    }
130
131    fn expire_at(&self) -> u32 {
132        self.expire_at.timestamp
133    }
134
135    fn hash(&self) -> &[u8] {
136        self.hash.as_slice()
137    }
138
139    fn sign(&self, signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH]) -> Result<SignedMessage> {
140        let mut payload = self.payload.clone();
141        payload.append_raw(signature, signature.len() * 8)?;
142
143        let mut message = self.message.clone();
144        message.set_body(SliceData::load_builder(payload)?);
145
146        Ok(SignedMessage {
147            message,
148            expire_at: self.expire_at(),
149        })
150    }
151
152    fn sign_with_pruned_payload(
153        &self,
154        signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH],
155        prune_after_depth: u16,
156    ) -> Result<SignedMessage> {
157        let mut payload = self.payload.clone();
158        payload.append_raw(signature, signature.len() * 8)?;
159        let body = payload.into_cell()?;
160
161        let mut message = self.message.clone();
162        message.set_body(prune_deep_cells(&body, prune_after_depth)?);
163
164        Ok(SignedMessage {
165            message,
166            expire_at: self.expire_at(),
167        })
168    }
169}

202/// `WalletV5` init data
203#[derive(Clone)]
204pub struct InitData {
205    pub is_signature_allowed: bool,
206    pub seqno: u32,
207    pub wallet_id: u32,
208    pub public_key: UInt256,
209    pub extensions: Option<Cell>,
210}
211
212impl InitData {
213    pub fn public_key(&self) -> &[u8; 32] {
214        self.public_key.as_slice()
215    }
216
217    pub fn from_key(key: &PublicKey) -> Self {
218        Self {
219            is_signature_allowed: false,
220            seqno: 0,
221            wallet_id: 0,
222            public_key: key.as_bytes().into(),
223            extensions: Default::default(),
224        }
225    }
226
227    pub fn with_is_signature_allowed(mut self, is_allowed: bool) -> Self {
228        self.is_signature_allowed = is_allowed;
229        self
230    }
231
232    pub fn with_wallet_id(mut self, id: u32) -> Self {
233        self.wallet_id = id;
234        self
235    }
236
237    pub fn compute_addr(&self, workchain_id: i8) -> Result<MsgAddressInt> {
238        let init_state = self.make_state_init()?.serialize()?;
239        let hash = init_state.repr_hash();
240        Ok(MsgAddressInt::AddrStd(MsgAddrStd {
241            anycast: None,
242            workchain_id,
243            address: hash.into(),
244        }))
245    }
246
247    pub fn make_state_init(&self) -> Result<ton_block::StateInit> {
248        Ok(ton_block::StateInit {
249            code: Some(nekoton_contracts::wallets::code::wallet_v5r1()),
250            data: Some(self.serialize()?),
251            ..Default::default()
252        })
253    }
254
255    pub fn serialize(&self) -> Result<Cell> {
256        let mut data = BuilderData::new();
257        data.append_bit_bool(self.is_signature_allowed)?
258            .append_u32(self.seqno)?
259            .append_u32(self.wallet_id)?
260            .append_raw(self.public_key.as_slice(), 256)?;
261
262        if let Some(extensions) = &self.extensions {
263            data.append_bit_one()?
264                .checked_append_reference(extensions.clone())?;
265        } else {
266            data.append_bit_zero()?;
267        }
268
269        data.into_cell()
270    }
271
272    pub fn make_transfer_payload(
273        &self,
274        gifts: impl IntoIterator<Item = Gift>,
275        expire_at: u32,
276    ) -> Result<(UInt256, BuilderData)> {
277        // Check if signatures are allowed
278        if !self.is_signature_allowed {
279            return if self.extensions.is_none() {
280                Err(WalletV5Error::WalletLocked.into())
281            } else {
282                Err(WalletV5Error::SignaturesDisabled.into())
283            };
284        }
285
286        let mut payload = BuilderData::new();
287
288        // insert prefix
289        payload
290            .append_u32(SIGNED_EXTERNAL_PREFIX)?
291            .append_u32(self.wallet_id)?
292            .append_u32(expire_at)?
293            .append_u32(self.seqno)?;
294
295        let mut actions = ton_block::OutActions::new();
296
297        for gift in gifts {
298            let mut internal_message =
299                ton_block::Message::with_int_header(ton_block::InternalMessageHeader {
300                    ihr_disabled: true,
301                    bounce: gift.bounce,
302                    dst: gift.destination,
303                    value: ton_block::CurrencyCollection::from_grams(ton_block::Grams::new(
304                        gift.amount,
305                    )?),
306                    ..Default::default()
307                });
308
309            if let Some(body) = gift.body {
310                internal_message.set_body(body);
311            }
312
313            if let Some(state_init) = gift.state_init {
314                internal_message.set_state_init(state_init);
315            }
316
317            let action = ton_block::OutAction::SendMsg {
318                mode: gift.flags,
319                out_msg: internal_message,
320            };
321
322            actions.push_back(action);
323        }
324
325        payload.append_bit_one()?;
326        payload.checked_append_reference(actions.serialize()?)?;
327
328        // has_other_actions
329        payload.append_bit_zero()?;
330
331        let hash = payload.clone().into_cell()?.repr_hash();
332
333        Ok((hash, payload))
334    }
335}

337impl TryFrom<&Cell> for InitData {
338    type Error = anyhow::Error;
339
340    fn try_from(data: &Cell) -> Result<Self, Self::Error> {
341        let mut cs = SliceData::load_cell_ref(data)?;
342        Ok(Self {
343            is_signature_allowed: cs.get_next_bit()?,
344            seqno: cs.get_next_u32()?,
345            wallet_id: cs.get_next_u32()?,
346            public_key: UInt256::from_be_bytes(&cs.get_next_bytes(32)?),
347            extensions: cs.get_next_dictionary()?,
348        })
349    }
350}
