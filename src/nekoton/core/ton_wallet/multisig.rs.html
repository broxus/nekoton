36pub fn prepare_deploy(
37    clock: &dyn Clock,
38    public_key: &PublicKey,
39    multisig_type: MultisigType,
40    workchain: i8,
41    expiration: Expiration,
42    params: DeployParams<'_>,
43) -> Result<Box<dyn UnsignedMessage>> {
44    let state_init = prepare_state_init(public_key, multisig_type);
45    let hash = state_init.hash().trust_me();
46
47    let dst = MsgAddressInt::AddrStd(ton_block::MsgAddrStd {
48        anycast: None,
49        workchain_id: workchain,
50        address: hash.into(),
51    });
52
53    let mut message =
54        ton_block::Message::with_ext_in_header(ton_block::ExternalInboundMessageHeader {
55            dst,
56            ..Default::default()
57        });
58
59    message.set_state_init(state_init);
60
61    let owners = params
62        .owners
63        .iter()
64        .map(|public_key| UInt256::from(public_key.as_bytes()))
65        .collect::<Vec<UInt256>>();
66
67    let is_new_multisig = multisig_type.is_multisig2();
68    let function = if is_new_multisig {
69        nekoton_contracts::wallets::multisig2::constructor()
70    } else if params.expiration_time.is_none() {
71        nekoton_contracts::wallets::multisig::constructor()
72    } else {
73        return Err(MultisigError::CustomExpirationTimeNotSupported.into());
74    };
75
76    let (function, input) = {
77        let mut message = MessageBuilder::new(function)
78            .arg(owners)
79            .arg(params.req_confirms);
80        if is_new_multisig {
81            message = message.arg(params.expiration_time.unwrap_or(DEFAULT_LIFETIME));
82        }
83        message.build()
84    };
85
86    make_labs_unsigned_message(
87        clock,
88        message,
89        expiration,
90        public_key,
91        Cow::Borrowed(function),
92        input,
93    )
94}

114pub fn prepare_transfer(
115    clock: &dyn Clock,
116    multisig_type: MultisigType,
117    public_key: &PublicKey,
118    has_multiple_owners: bool,
119    address: MsgAddressInt,
120    gift: Gift,
121    expiration: Expiration,
122) -> Result<TransferAction> {
123    let is_new_multisig = multisig_type.is_multisig2();
124
125    let (function, input) = if has_multiple_owners || is_new_multisig && gift.state_init.is_some() {
126        let all_balance = match MessageFlags::try_from(gift.flags) {
127            Ok(MessageFlags::Normal) => false,
128            Ok(MessageFlags::AllBalance) => true,
129            _ => return Err(MultisigError::UnsupportedFlagsSet.into()),
130        };
131
132        let function = if is_new_multisig {
133            nekoton_contracts::wallets::multisig2::submit_transaction()
134        } else {
135            nekoton_contracts::wallets::multisig::submit_transaction()
136        };
137
138        let message = MessageBuilder::new(function)
139            .arg(gift.destination)
140            .arg(BigUint128(gift.amount.into()))
141            .arg(gift.bounce)
142            .arg(all_balance)
143            .arg(gift.body.unwrap_or_default().into_cell());
144
145        if is_new_multisig {
146            message
147                .arg(
148                    gift.state_init
149                        .map(|state_init| state_init.serialize())
150                        .transpose()?,
151                )
152                .build()
153        } else {
154            message.build()
155        }
156    } else {
157        let function = if is_new_multisig {
158            nekoton_contracts::wallets::multisig2::send_transaction()
159        } else {
160            nekoton_contracts::wallets::multisig::send_transaction()
161        };
162        MessageBuilder::new(function)
163            .arg(gift.destination)
164            .arg(BigUint128(gift.amount.into()))
165            .arg(gift.bounce)
166            .arg(gift.flags)
167            .arg(gift.body.unwrap_or_default().into_cell())
168            .build()
169    };
170
171    make_ext_message(clock, public_key, address, expiration, function, input)
172        .map(TransferAction::Sign)
173}

354    match code_hash.as_slice() {
355        s if s == SAFE_MULTISIG_WALLET_HASH => Some(MultisigType::SafeMultisigWallet),

357        s if s == SETCODE_MULTISIG_WALLET_HASH => Some(MultisigType::SetcodeMultisigWallet),
358        s if s == BRIDGE_MULTISIG_WALLET_HASH => Some(MultisigType::BridgeMultisigWallet),
359        s if s == SETCODE_MULTISIG_WALLET_24H_HASH => Some(MultisigType::SetcodeMultisigWallet24h),
360        s if s == SURF_WALLET_HASH => Some(MultisigType::SurfWallet),

362        s if s == MULTISIG2_1_HASH => Some(MultisigType::Multisig2_1),

477pub fn get_custodians(
478    clock: &dyn Clock,
479    multisig_type: MultisigType,
480    account_stuff: Cow<'_, ton_block::AccountStuff>,
481) -> Result<Vec<UInt256>> {
482    let function = if multisig_type.is_multisig2() {
483        nekoton_contracts::wallets::multisig2::get_custodians()
484    } else {
485        nekoton_contracts::wallets::multisig::get_custodians()
486    };
487    run_local(clock, function, account_stuff.into_owned())
488        .and_then(parse_multisig_contract_custodians)
489}
490
491fn parse_multisig_contract_custodians(tokens: Vec<ton_abi::Token>) -> Result<Vec<UInt256>> {
492    let array = match tokens.into_unpacker().unpack_next() {
493        Ok(ton_abi::TokenValue::Array(_, tokens)) => tokens,
494        _ => return Err(UnpackerError::InvalidAbi.into()),
495    };
496
497    let mut custodians = array
498        .into_iter()
499        .map(|item| item.unpack())
500        .collect::<Result<Vec<nekoton_contracts::wallets::multisig::MultisigCustodian>, _>>()?;
501
502    custodians.sort_by(|a, b| a.index.cmp(&b.index));
503
504    Ok(custodians.into_iter().map(|item| item.pubkey).collect())
505}

507pub fn find_pending_transaction(
508    clock: &dyn Clock,
509    multisig_type: MultisigType,
510    account_stuff: Cow<'_, ton_block::AccountStuff>,
511    pending_transaction_id: u64,
512) -> Result<bool> {
513    #[derive(Copy, Clone, UnpackAbi)]
514    pub struct MultisigTransactionId {
515        #[abi(uint64)]
516        pub id: u64,
517    }
518
519    let function = if multisig_type.is_multisig2() {
520        nekoton_contracts::wallets::multisig2::get_transactions()
521    } else {
522        nekoton_contracts::wallets::multisig::get_transactions()
523    };
524
525    let tokens = run_local(clock, function, account_stuff.into_owned())?;
526
527    let array = match tokens.into_unpacker().unpack_next() {
528        Ok(ton_abi::TokenValue::Array(_, tokens)) => tokens,
529        _ => return Err(UnpackerError::InvalidAbi.into()),
530    };
531
532    for item in array {
533        let MultisigTransactionId { id } = item.unpack()?;
534        if pending_transaction_id == id {
535            return Ok(true);
536        }
537    }
538    Ok(false)
539}

541pub fn find_pending_update(
542    clock: &dyn Clock,
543    multisig_type: MultisigType,
544    account_stuff: Cow<'_, ton_block::AccountStuff>,
545    update_id: u64,
546) -> Result<Option<UpdatedParams>> {
547    use nekoton_contracts::wallets::multisig2;
548
549    let function = match multisig_type {
550        MultisigType::Multisig2 => multisig2::v2_0::get_update_requests(),
551        MultisigType::Multisig2_1 => multisig2::v2_1::get_update_requests(),
552        _ => return Ok(None),
553    };
554
555    let tokens = run_local(clock, function, account_stuff.into_owned())?;
556
557    let array = match tokens.into_unpacker().unpack_next() {
558        Ok(ton_abi::TokenValue::Array(_, tokens)) => tokens,
559        _ => return Err(UnpackerError::InvalidAbi.into()),
560    };
561
562    for item in array {
563        let update: multisig2::UpdateTransaction = item.unpack()?;
564        if update_id == update.id {
565            return Ok(Some(UpdatedParams {
566                new_code_hash: update.new_code_hash,
567                new_custodians: update.new_custodians,
568                new_req_confirms: update.new_req_confirms,
569                new_lifetime: update.new_lifetime,
570            }));
571        }
572    }
573
574    Ok(None)
575}

585pub fn get_pending_transactions(
586    clock: &dyn Clock,
587    multisig_type: MultisigType,
588    account_stuff: Cow<'_, ton_block::AccountStuff>,
589    custodians: &[UInt256],
590) -> Result<Vec<MultisigPendingTransaction>> {
591    let function = if multisig_type.is_multisig2() {
592        nekoton_contracts::wallets::multisig2::get_transactions()
593    } else {
594        nekoton_contracts::wallets::multisig::get_transactions()
595    };
596    run_local(clock, function, account_stuff.into_owned()).and_then(|tokens| {
597        let array = match tokens.into_unpacker().unpack_next() {
598            Ok(ton_abi::TokenValue::Array(_, tokens)) => tokens,
599            _ => return Err(UnpackerError::InvalidAbi.into()),
600        };
601
602        let transactions = array
603            .into_iter()
604            .map(|item| Ok(extend_pending_transaction(item.unpack()?, custodians)))
605            .collect::<UnpackerResult<Vec<MultisigPendingTransaction>>>()?;
606
607        Ok(transactions)
608    })
609}

611pub fn get_pending_updates(
612    clock: &dyn Clock,
613    multisig_type: MultisigType,
614    account_stuff: Cow<'_, ton_block::AccountStuff>,
615    custodians: &[UInt256],
616) -> Result<Vec<MultisigPendingUpdate>> {
617    use nekoton_contracts::wallets::multisig2;
618
619    let function = match multisig_type {
620        MultisigType::Multisig2 => multisig2::v2_0::get_update_requests(),
621        MultisigType::Multisig2_1 => multisig2::v2_1::get_update_requests(),
622        _ => return Ok(Vec::new()),
623    };
624
625    run_local(clock, function, account_stuff.into_owned()).and_then(|tokens| {
626        let array = match tokens.into_unpacker().unpack_next() {
627            Ok(ton_abi::TokenValue::Array(_, tokens)) => tokens,
628            _ => return Err(UnpackerError::InvalidAbi.into()),
629        };
630
631        let updates = array
632            .into_iter()
633            .map(|item| Ok(extend_pending_update(item.unpack()?, custodians)))
634            .collect::<UnpackerResult<Vec<MultisigPendingUpdate>>>()?;
635
636        Ok(updates)
637    })
638}

640fn extend_pending_transaction(
641    tx: nekoton_contracts::wallets::multisig::MultisigTransaction,
642    custodians: &[UInt256],
643) -> MultisigPendingTransaction {
644    let confirmations = custodians
645        .iter()
646        .enumerate()
647        .filter(|(i, _)| (0b1 << i) & tx.confirmation_mask != 0)
648        .map(|(_, item)| *item)
649        .collect::<Vec<UInt256>>();
650
651    MultisigPendingTransaction {
652        id: tx.id,
653        confirmations,
654        signs_required: tx.signs_required,
655        signs_received: tx.signs_received,
656        creator: tx.creator,
657        index: tx.index,
658        dest: tx.dest,
659        value: tx.value.into(),
660        send_flags: tx.send_flags,
661        payload: tx.payload,
662        bounce: tx.bounce,
663    }
664}

666fn extend_pending_update(
667    tx: nekoton_contracts::wallets::multisig2::UpdateTransaction,
668    custodians: &[UInt256],
669) -> MultisigPendingUpdate {

672        .enumerate()
673        .filter(|(i, _)| (0b1 << i) & tx.confirmations_mask != 0)
674        .map(|(_, item)| *item)
675        .collect::<Vec<UInt256>>();
676
677    MultisigPendingUpdate {
678        id: tx.id,
679        confirmations,
680        signs_received: tx.signs,
681        creator: tx.creator,
682        index: tx.index,
683        new_code_hash: tx.new_code_hash,
684        new_custodians: tx.new_custodians,
685        new_req_confirms: tx.new_req_confirms,
686        new_lifetime: tx.new_lifetime,
687    }
