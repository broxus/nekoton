29pub fn request_transactions<'a>(
30    transport: &'a dyn Transport,
31    address: &'a MsgAddressInt,
32    from_lt: u64,
33    until_lt: Option<u64>,
34    initial_count: u8,
35    limit: Option<usize>,
36) -> impl Stream<Item = Result<Vec<RawTransaction>>> + 'a {
37    let initial_count = u8::min(initial_count, transport.info().max_transactions_per_fetch);
38    let fut = transport.get_transactions(address, from_lt, initial_count);
39
40    LatestTransactions {
41        address,
42        from_lt,
43        until_lt,
44        transport,
45        fut: Some(fut),
46        initial_count,
47        total_fetched: 0,
48        limit,
49    }
50}

80pub fn parse_block(
81    address: &MsgAddressInt,
82    contract_state: &ContractState,
83    block: &ton_block::Block,
84) -> Result<ParsedBlock> {
85    use ton_block::{Deserializable, HashmapAugType};
86    use ton_types::HashmapType;
87
88    let info = block
89        .info
90        .read_struct()
91        .map_err(|_| BlockParsingError::InvalidBlockStructure)?;
92
93    let account_block = match block
94        .extra
95        .read_struct()
96        .and_then(|extra| extra.read_account_blocks())
97        .and_then(|account_blocks| {
98            account_blocks.get_with_aug(&ton_types::UInt256::from_be_bytes(
99                &address.address().get_bytestring(0),
100            ))
101        }) {
102        Ok(Some((extra, _))) => extra,
103        _ => return Ok(ParsedBlock::empty(info.gen_utime().as_u32())),
104    };
105
106    let mut balance = contract_state.balance as i128;
107    let mut new_transactions = Vec::new();
108
109    let mut last_lt = contract_state.last_lt;
110    let mut latest_transaction_id: Option<TransactionId> = None;
111    let mut is_deployed = contract_state.is_deployed;
112
113    for item in account_block.transactions().iter() {
114        let result = item.and_then(|(_, value)| {
115            let cell = value.into_cell().reference(0)?;
116            let hash = cell.repr_hash();
117
118            ton_block::Transaction::construct_from_cell(cell)
119                .map(|data| RawTransaction { hash, data })
120        });
121        let transaction = match result {
122            Ok(transaction) => transaction,
123            Err(_) => continue,
124        };
125
126        balance += compute_balance_change(&transaction.data);
127
128        is_deployed = transaction.data.end_status == ton_block::AccountStatus::AccStateActive;
129
130        if matches!(&latest_transaction_id, Some(id) if transaction.data.lt > id.lt) {
131            latest_transaction_id = Some(TransactionId {
132                lt: transaction.data.lt,
133                hash: transaction.hash,
134            })
135        }
136
137        last_lt = std::cmp::max(last_lt, compute_account_lt(&transaction.data));
138        new_transactions.push(transaction);
139    }
140
141    let new_contract_state = ContractState {
142        last_lt,
143        balance: balance.try_into().unwrap_or_default(),
144        gen_timings: GenTimings::Known {
145            gen_lt: info.end_lt(),
146            gen_utime: info.gen_utime().as_u32(),
147        },
148        last_transaction_id: latest_transaction_id
149            .map(LastTransactionId::Exact)
150            .or(contract_state.last_transaction_id),
151        is_deployed,
152        code_hash: contract_state.code_hash, // NOTE: code hash update is not visible
153    };
154
155    let new_transactions =
156        if let (Some(first), Some(last)) = (new_transactions.first(), new_transactions.last()) {
157            Some(TransactionsBatchInfo {
158                min_lt: first.data.lt, // transactions in block info are in ascending order
159                max_lt: last.data.lt,
160                batch_type: TransactionsBatchType::New,
161            })
162        } else {
163            None
164        }
165        .map(|batch_info| (new_transactions, batch_info));
166
167    Ok(ParsedBlock::with_data(
168        info.gen_utime().as_u32(),
169        new_contract_state,
170        new_transactions,
171    ))
172}

182struct LatestTransactions<'a> {
183    address: &'a MsgAddressInt,
184    from_lt: u64,
185    until_lt: Option<u64>,
186    transport: &'a dyn Transport,
187    fut: Option<TransactionsFut<'a>>,
188    initial_count: u8,
189    total_fetched: usize,
190    limit: Option<usize>,
191}

195#[cfg(feature = "non_threadsafe")]
196type TransactionsFut<'a> = Pin<Box<dyn Future<Output = Result<Vec<RawTransaction>>> + 'a>>;

198impl Stream for LatestTransactions<'_> {
199    type Item = Result<Vec<RawTransaction>>;
200
201    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
202        loop {
203            // poll `get_transactions` future
204            let new_transactions = match self.fut.take() {
205                Some(mut fut) => match fut.poll_unpin(cx) {
206                    Poll::Ready(result) => result,
207                    Poll::Pending => {
208                        self.fut = Some(fut);
209                        return Poll::Pending;
210                    }
211                },
212                None => return Poll::Ready(None),
213            };
214
215            let mut new_transactions = match new_transactions {
216                Ok(transactions) => transactions,
217                // return error without resetting future
218                Err(e) => return Poll::Ready(Some(Err(e))),
219            };
220
221            // ensure that transactions are sorted in reverse order (from the latest lt)
222            new_transactions.sort_by_key(|tx| std::cmp::Reverse(tx.data.lt));
223
224            // get next lt from the unfiltered response to continue
225            // fetching transactions if filter produced empty array
226            let next_lt_from_response = match new_transactions.last() {
227                Some(last) => last.data.prev_trans_lt,
228                // early return on empty response
229                None => return Poll::Ready(None),
230            };
231            let mut possibly_has_more = next_lt_from_response > 0;
232
233            let mut truncated = false;
234            if let Some(first_tx) = new_transactions.first() {
235                if first_tx.data.lt > self.from_lt {
236                    // retain only elements in range (until_lt; from_lt]
237                    // NOTE: `until_lt < from_lt`
238                    let until_lt = self.until_lt.unwrap_or_default();
239                    let range = (until_lt + 1)..=self.from_lt;
240
241                    new_transactions.retain(|item| {
242                        possibly_has_more &= item.data.lt > until_lt;
243                        range.contains(&item.data.lt)
244                    });
245                    truncated = true;
246                }
247            }
248
249            if !truncated {
250                if let Some(until_lt) = self.until_lt {
251                    if let Some(len) = new_transactions
252                        .iter()
253                        .position(|tx| tx.data.lt <= until_lt)
254                    {
255                        new_transactions.truncate(len);
256                        possibly_has_more = false;
257                    }
258                }
259            }
260
261            // get batch info
262            let last = match new_transactions.last() {
263                Some(last) => last,
264                None if possibly_has_more => {
265                    self.fut = Some(self.transport.get_transactions(
266                        self.address,
267                        next_lt_from_response,
268                        self.initial_count,
269                    ));
270                    continue;
271                }
272                None => return Poll::Ready(None),
273            };
274
275            // set next batch bound
276            self.from_lt = last.data.prev_trans_lt;
277
278            // check if there are no transactions left or all transactions were requested
279            if last.data.prev_trans_lt == 0
280                || matches!(self.until_lt, Some(until_lt) if last.data.prev_trans_lt <= until_lt)
281            {
282                return Poll::Ready(Some(Ok(new_transactions)));
283            }
284
285            // update counters
286            self.total_fetched += new_transactions.len();
287
288            let next_count = match self.limit {
289                Some(limit) if self.total_fetched >= limit => {
290                    return Poll::Ready(Some(Ok(new_transactions)))
291                }
292                Some(limit) => usize::min(
293                    limit - self.total_fetched,
294                    self.transport.info().max_transactions_per_fetch as usize,
295                ) as u8,
296                None => self.transport.info().max_transactions_per_fetch,
297            };
298
299            // If there are some unprocessed transactions left we should request remaining
300            self.fut = Some(self.transport.get_transactions(
301                self.address,
302                last.data.prev_trans_lt,
303                next_count,
304            ));
305
306            // Return result
307            return Poll::Ready(Some(Ok(new_transactions)));
308        }
309    }
310}

319pub trait PendingTransactionsExt {
320    fn add_message(
321        &mut self,
322        target: &MsgAddressInt,
323        message: &ton_block::Message,
324        ctx: MessageContext,
325    ) -> Result<PendingTransaction>;
326
327    fn cancel(&mut self, pending_transaction: &PendingTransaction);
328}
329
330impl PendingTransactionsExt for Vec<PendingTransaction> {
331    fn add_message(
332        &mut self,
333        target: &MsgAddressInt,
334        message: &ton_block::Message,
335        ctx: MessageContext,
336    ) -> Result<PendingTransaction> {
337        let src = match message.header() {
338            ton_block::CommonMsgInfo::ExtInMsgInfo(header) => {
339                if &header.dst == target {
340                    None
341                } else {
342                    return Err(AccountSubscriptionError::InvalidMessageDestination.into());
343                }
344            }
345            _ => return Err(AccountSubscriptionError::InvalidMessageType.into()),
346        };
347
348        let pending_transaction = PendingTransaction {
349            message_hash: message.serialize()?.repr_hash(),
350            src,
351            latest_lt: ctx.latest_lt,
352            created_at: ctx.created_at,
353            expire_at: ctx.expire_at,
354        };
355
356        self.push(pending_transaction.clone());
357        Ok(pending_transaction)
358    }
359
360    fn cancel(&mut self, pending_transaction: &PendingTransaction) {
361        if let Some(i) = self.iter().position(|item| item.eq(pending_transaction)) {
362            self.remove(i);
363        }
364    }
365}

367pub fn make_labs_unsigned_message(
368    clock: &dyn Clock,
369    message: ton_block::Message,
370    expiration: Expiration,
371    public_key: &PublicKey,
372    function: Cow<'static, ton_abi::Function>,
373    input: Vec<ton_abi::Token>,
374) -> Result<Box<dyn UnsignedMessage>> {
375    let time = clock.now_ms_u64();
376    let (expire_at, header) = default_headers(time, expiration, public_key);
377
378    let (payload, hash) =
379        function.create_unsigned_call(&header, &input, false, true, message.dst())?;
380
381    Ok(Box::new(LabsUnsignedMessage {
382        function,
383        header,
384        input,
385        payload,
386        hash,
387        expire_at,
388        message,
389    }))
390}

392#[derive(Clone)]
393struct LabsUnsignedMessage {
394    function: Cow<'static, ton_abi::Function>,
395    header: HeadersMap,
396    input: Vec<ton_abi::Token>,
397    payload: ton_types::BuilderData,
398    hash: ton_types::UInt256,
399    expire_at: ExpireAt,
400    message: ton_block::Message,
401}
402
403impl UnsignedMessage for LabsUnsignedMessage {
404    fn refresh_timeout(&mut self, clock: &dyn Clock) {
405        let time = clock.now_ms_u64();
406
407        if !self.expire_at.refresh_from_millis(time) {
408            return;
409        }
410
411        *self.header.get_mut("time").trust_me() = ton_abi::TokenValue::Time(time);
412        *self.header.get_mut("expire").trust_me() = ton_abi::TokenValue::Expire(self.expire_at());
413
414        let (payload, hash) = self
415            .function
416            .create_unsigned_call(&self.header, &self.input, false, true, self.message.dst())
417            .trust_me();
418        self.payload = payload;
419        self.hash = hash;
420    }
421
422    fn expire_at(&self) -> u32 {
423        self.expire_at.timestamp
424    }
425
426    fn hash(&self) -> &[u8] {
427        self.hash.as_slice()
428    }
429
430    fn sign(&self, signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH]) -> Result<SignedMessage> {
431        let payload = self.payload.clone();
432        let payload = ton_abi::Function::fill_sign(
433            &self.function.abi_version,
434            Some(signature),
435            None,
436            payload,
437        )
438        .and_then(ton_types::SliceData::load_builder)?;
439
440        let mut message = self.message.clone();
441        message.set_body(payload);
442
443        Ok(SignedMessage {
444            message,
445            expire_at: self.expire_at(),
446        })
447    }
448
449    fn sign_with_pruned_payload(
450        &self,
451        signature: &[u8; ed25519_dalek::SIGNATURE_LENGTH],
452        prune_after_depth: u16,
453    ) -> Result<SignedMessage> {
454        let payload = self.payload.clone();
455        let payload = ton_abi::Function::fill_sign(
456            &self.function.abi_version,
457            Some(signature),
458            None,
459            payload,
460        )?
461        .into_cell()?;
462
463        let mut message = self.message.clone();
464        message.set_body(prune_deep_cells(&payload, prune_after_depth)?);
465
466        Ok(SignedMessage {
467            message,
468            expire_at: self.expire_at(),
469        })
470    }
471}

473pub fn default_headers(
474    time: u64,
475    expiration: Expiration,
476    public_key: &PublicKey,
477) -> (ExpireAt, HeadersMap) {
478    let expire_at = ExpireAt::new_from_millis(expiration, time);
479
480    let mut header = HashMap::with_capacity(3);
481    header.insert("time".to_string(), ton_abi::TokenValue::Time(time));
482    header.insert(
483        "expire".to_string(),
484        ton_abi::TokenValue::Expire(expire_at.timestamp),
485    );
486    header.insert(
487        "pubkey".to_string(),
488        ton_abi::TokenValue::PublicKey(Some(*public_key)),
489    );
490
491    (expire_at, header)
492}

496pub async fn update_library_cell(
497    transport: &dyn Transport,
498    state: &mut AccountState,
499) -> Result<()> {
500    if let AccountState::AccountActive { ref mut state_init } = state {
501        if let Some(cell) = &state_init.code {
502            if cell.cell_type() == CellType::LibraryReference {
503                let mut slice_data = SliceData::load_cell(cell.clone())?;
504
505                // Read Library Cell Tag
506                let tag = slice_data.get_next_byte()?;
507                assert_eq!(tag, 2);
508
509                // Read Code Hash
510                let mut hash = UInt256::default();
511                hash.read_from(&mut slice_data)?;
512
513                let cell = transport.get_library_cell(&hash).await?;
514                if let Some(cell) = cell {
515                    state_init.set_code(cell);
516                }
517            }
518        }
519    }
520
521    Ok(())
