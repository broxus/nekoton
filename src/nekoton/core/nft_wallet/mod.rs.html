31impl NftCollection {
32    pub async fn new(
33        clock: &dyn Clock,
34        transport: Arc<dyn Transport>,
35        collection_address: MsgAddressInt,
36    ) -> Result<NftCollection> {
37        let state = match transport.get_contract_state(&collection_address).await? {
38            RawContractState::Exists(state) => state,
39            RawContractState::NotExists { .. } => return Err(NftError::ContractNotExist.into()),
40        };
41
42        let contract = CollectionContractState(&state);
43
44        let interfaces = contract.check_collection_supported_interfaces(clock)?;
45        if !interfaces.tip4_3 {
46            return Err(NftError::InvalidCollectionContract.into());
47        }
48
49        let index_code = contract.resolve_collection_index_code(clock)?;
50
51        let json_info = interfaces
52            .tip4_2
53            .then(|| tip4_2::MetadataContract(state.as_context(clock)).get_json())
54            .transpose()?;
55
56        Ok(NftCollection {
57            transport,
58            collection_address,
59            state,
60            index_code,
61            json_info,
62        })
63    }

66        &self.collection_address
67    }

70        &self.index_code
71    }
72
73    pub fn json_info(&self) -> &Option<String> {
74        &self.json_info
75    }

77    pub fn compute_collection_code_hash(&self, owner: &MsgAddressInt) -> Result<UInt256> {
78        CollectionContractState(&self.state)
79            .get_collection_code_hash(owner, self.index_code.clone())
80    }

82    pub async fn get_nft_index_contracts(
83        &self,
84        owner: &MsgAddressInt,
85        limit: u8,
86        continuation: Option<MsgAddressInt>,

91            .await
92    }

133    async fn subscribe(
134        nft_address: &MsgAddressInt,
135        transport: Arc<dyn Transport>,
136        clock: Arc<dyn Clock>,
137        handler: Arc<dyn NftSubscriptionHandler>,
138    ) -> Result<Nft> {
139        let state = match transport.get_contract_state(nft_address).await? {
140            RawContractState::Exists(state) => state,
141            RawContractState::NotExists { .. } => return Err(NftError::ContractNotExist.into()),
142        };
143        let nft_state = NftContractState(&state);
144
145        let interfaces = match nft_state.check_supported_interfaces(clock.as_ref())? {
146            Some(interfaces) => interfaces,
147            None => return Err(NftError::InvalidNftContact.into()),
148        };
149
150        // NOTE: Assume that TIP4.1 support is mandatory
151        let mut info = nft_state.get_info(clock.as_ref())?;
152        let json_info = if interfaces.tip4_2 {
153            Some(nft_state.get_json(clock.as_ref())?)
154        } else {
155            None
156        };
157
158        let contract_subscription = ContractSubscription::subscribe(
159            clock.clone(),
160            transport,
161            nft_address.clone(),
162            &mut make_contract_state_handler(
163                clock.as_ref(),
164                &mut info.owner,
165                &mut info.manager,
166                None,
167            ),
168            Some(&mut make_transactions_handler(handler.as_ref())),
169        )
170        .await?;
171
172        handler.on_manager_changed(info.manager.clone());
173        handler.on_owner_changed(info.owner.clone());
174
175        Ok(Self {
176            clock,
177            address: nft_address.clone(),
178            collection_address: info.collection,
179            owner: info.owner,
180            manager: info.manager,
181            interfaces,
182            json_info,
183            contract_subscription,
184            handler,
185        })
186    }

216    pub fn prepare_transfer(
217        &self,
218        to: MsgAddressInt,
219        send_gas_to: MsgAddressInt,
220        callbacks: BTreeMap<MsgAddressInt, NftCallbackPayload>,
221    ) -> Result<InternalMessage> {
222        const ATTACHED_AMOUNT: u128 = 1_000_000_000; // 1 TON
223        let (function, input) = MessageBuilder::new(nft_contract::transfer())
224            .arg(to)
225            .arg(send_gas_to)
226            .arg(callbacks)
227            .build();
228
229        let body = function
230            .encode_internal_input(&input)
231            .and_then(ton_types::SliceData::load_builder)?;
232
233        Ok(InternalMessage {
234            source: Some(self.owner.clone()),
235            destination: self.address().clone(),
236            amount: ATTACHED_AMOUNT,
237            bounce: true,
238            body,
239        })
240    }

328pub trait NftSubscriptionHandler: Send + Sync {
329    /// Called when found transaction which is relative with one of the pending transactions
330    fn on_message_sent(
331        &self,
332        pending_transaction: PendingTransaction,
333        transaction: Option<Transaction>,
334    );
335
336    /// Called when no transactions produced for the specific message before some expiration time
337    fn on_message_expired(&self, pending_transaction: PendingTransaction);
338
339    fn on_manager_changed(&self, manager: MsgAddressInt) {
340        let _ = manager;
341    }
342
343    fn on_owner_changed(&self, owner: MsgAddressInt) {
344        let _ = owner;
345    }
346
347    /// Called every time new transactions are detected.
348    /// - When new block found
349    /// - When manually requesting the latest transactions (can be called several times)
350    /// - When preloading transactions
351    fn on_transactions_found(
352        &self,
353        transactions: Vec<TransactionWithData<NftTransaction>>,
354        batch_info: TransactionsBatchInfo,
355    ) {
356        let _ = transactions;
357        let _ = batch_info;
358    }
359}
360
361fn make_contract_state_handler<'a>(
362    clock: &'a dyn Clock,
363    owner: &'a mut MsgAddressInt,
364    manager: &'a mut MsgAddressInt,
365    handler: Option<&'a dyn NftSubscriptionHandler>,
366) -> impl FnMut(&RawContractState) + 'a {
367    move |contract_state| {
368        if let RawContractState::Exists(state) = contract_state {
369            if let Ok(info) = NftContractState(state).get_info(clock) {
370                let mut owner_changed = false;
371                if owner != &info.owner {
372                    *owner = info.owner;
373                    owner_changed = true;
374                }
375
376                let mut manager_changed = false;
377                if manager != &info.manager {
378                    *manager = info.manager;
379                    manager_changed = true;
380                }
381
382                if let Some(handler) = handler {
383                    if owner_changed {
384                        handler.on_owner_changed(owner.clone());
385                    }
386                    if manager_changed {
387                        handler.on_manager_changed(manager.clone());
388                    }
389                }
390            }
391        }
392    }
393}
394
395fn make_transactions_handler(
396    handler: &'_ dyn NftSubscriptionHandler,
397) -> impl FnMut(Vec<RawTransaction>, TransactionsBatchInfo) + '_ {
398    move |transactions, batch_info| {
399        let transactions = transactions
400            .into_iter()
401            .filter_map(
402                |transaction| match transaction.data.description.read_struct().ok()? {
403                    ton_block::TransactionDescr::Ordinary(description) => {
404                        let data = parse_nft_transaction(&transaction.data, &description);
405
406                        let transaction =
407                            Transaction::try_from((transaction.hash, transaction.data)).ok()?;
408
409                        Some(TransactionWithData { transaction, data })
410                    }
411                    _ => None,
412                },
413            )
414            .collect();
415
416        handler.on_transactions_found(transactions, batch_info)
417    }
418}
419
420fn make_message_sent_handler(
421    handler: &'_ dyn NftSubscriptionHandler,
422) -> impl FnMut(PendingTransaction, RawTransaction) + '_ {
423    move |pending_transaction, transaction| {
424        let transaction = Transaction::try_from((transaction.hash, transaction.data)).ok();
425        handler.on_message_sent(pending_transaction, transaction);
426    }
427}
428
429fn make_message_expired_handler(
430    handler: &'_ dyn NftSubscriptionHandler,
431) -> impl FnMut(PendingTransaction) + '_ {
432    move |pending_transaction| handler.on_message_expired(pending_transaction)
433}

435#[derive(Debug)]
436pub struct CollectionContractState<'a>(pub &'a ExistingContract);
437
438impl CollectionContractState<'_> {
439    pub fn check_collection_supported_interfaces(
440        &self,
441        clock: &dyn Clock,
442    ) -> Result<CollectionInterfaces> {
443        let ctx = self.0.as_context(clock);
444        let tip6_interface = tip6::SidContract(ctx);
445
446        let mut result = CollectionInterfaces::default();
447
448        if tip6_interface.supports_interface(tip4_3::collection_contract::INTERFACE_ID)? {
449            result.tip4_3 = true;
450            result.tip4_2 =
451                tip6_interface.supports_interface(tip4_2::metadata_contract::INTERFACE_ID)?;
452        }
453
454        Ok(result)
455    }
456
457    pub fn resolve_collection_index_code(&self, clock: &dyn Clock) -> Result<Cell> {
458        let ctx = self.0.as_context(clock);
459        tip4_3::CollectionContract(ctx).index_code()
460    }
461
462    pub fn get_collection_code_hash(
463        &self,
464        owner: &MsgAddressInt,
465        code_index: Cell,
466    ) -> Result<UInt256> {
467        let mut builder = BuilderData::new();
468
469        let owner_cell = owner.serialize()?;
470        let collection_cell = self.0.account.addr.serialize()?;
471
472        builder.append_raw(collection_cell.data(), collection_cell.bit_length())?;
473        builder.append_raw(owner_cell.data(), owner_cell.bit_length())?;
474
475        let mut nft = BuilderData::new();
476        nft.append_raw(NFT_STAMP, 24)?;
477
478        builder.checked_append_reference(nft.into_cell()?)?;
479
480        let salt = builder.into_cell()?;
481
482        let cell = nekoton_abi::set_code_salt(code_index, salt)?;
483        Ok(cell.hash(0))
484    }
485}

501#[derive(Debug)]
502pub struct NftContractState<'a>(pub &'a ExistingContract);
503
504impl NftContractState<'_> {
505    pub fn check_supported_interfaces(&self, clock: &dyn Clock) -> Result<Option<NftInterfaces>> {
506        let ctx = self.0.as_context(clock);
507        let tip6_interface = tip6::SidContract(ctx);
508
509        let mut result = NftInterfaces::default();
510        result.tip4_1 = tip6_interface.supports_interface(tip4_1::nft_contract::INTERFACE_ID)?;
511        result.tip4_2 =
512            tip6_interface.supports_interface(tip4_2::metadata_contract::INTERFACE_ID)?;
513        result.tip4_2_2 =
514            tip6_interface.supports_interface(tip4_2_2::metadata_contract::INTERFACE_ID)?;
515        result.tip4_3 = tip6_interface.supports_interface(tip4_3::nft_contract::INTERFACE_ID)?;
516
517        Ok((result.tip4_1 || result.tip4_3).then_some(result))
518    }
519
520    pub fn get_json(&self, clock: &dyn Clock) -> Result<String> {
521        let ctx = self.0.as_context(clock);
522        let tip4_2_interface = tip4_2::MetadataContract(ctx);
523        tip4_2_interface.get_json()
524    }

527        let ctx = self.0.as_context(clock);
528        let tip4_1_interface = tip4_1::NftContract(ctx);
