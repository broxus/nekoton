24pub fn parse_payload(payload: ton_types::SliceData) -> Option<KnownPayload> {
25    let function_id = read_function_id(&payload).ok()?;
26    if function_id == 0 {
27        return parse_comment_payload(payload).map(KnownPayload::Comment);
28    }
29
30    for version in [TokenWalletVersion::OldTip3v4, TokenWalletVersion::Tip3] {
31        let functions = TokenWalletFunctions::for_version(version);
32
33        if function_id == functions.transfer_to_wallet.input_id {
34            let inputs = functions
35                .transfer_to_wallet
36                .decode_input(payload, true, false)
37                .ok()?;
38
39            return TokenOutgoingTransfer::try_from((
40                InputMessage(inputs),
41                TransferType::ByTokenWalletAddress,
42                version,
43            ))
44            .map(KnownPayload::TokenOutgoingTransfer)
45            .ok();
46        } else if function_id == functions.transfer.input_id {
47            let inputs = functions.transfer.decode_input(payload, true, false).ok()?;
48
49            return TokenOutgoingTransfer::try_from((
50                InputMessage(inputs),
51                TransferType::ByOwnerWalletAddress,
52                version,
53            ))
54            .map(KnownPayload::TokenOutgoingTransfer)
55            .ok();
56        } else if function_id == functions.burn.input_id {
57            let inputs = functions.burn.decode_input(payload, true, false).ok()?;
58
59            return TokenSwapBack::try_from((InputMessage(inputs), version))
60                .map(KnownPayload::TokenSwapBack)
61                .ok();
62        }
63    }
64
65    None
66}

68pub fn parse_jetton_payload(payload: ton_types::SliceData) -> Option<KnownPayload> {
69    let mut payload = payload;
70
71    let opcode = payload.get_next_u32().ok()?;
72    if opcode != JETTON_TRANSFER_OPCODE {
73        return None;
74    }
75
76    let _query_id = payload.get_next_u64().ok()?;
77
78    let mut amount = ton_block::Grams::default();
79    amount.read_from(&mut payload).ok()?;
80
81    let mut dst_addr = MsgAddressInt::default();
82    dst_addr.read_from(&mut payload).ok()?;
83
84    Some(KnownPayload::JettonOutgoingTransfer(
85        JettonOutgoingTransfer {
86            to: dst_addr,
87            tokens: BigUint::from(amount.as_u128()),
88        },
89    ))
90}

92pub fn parse_transaction_additional_info(
93    tx: &ton_block::Transaction,
94    wallet_type: WalletType,
95) -> Option<TransactionAdditionalInfo> {
96    let in_msg = tx.in_msg.as_ref()?.read_struct().ok()?;
97
98    let int_header = match in_msg.header() {
99        ton_block::CommonMsgInfo::ExtInMsgInfo(_) => {
100            let (recipient, known_payload, method) = match wallet_type {
101                WalletType::WalletV3 | WalletType::HighloadWalletV2 | WalletType::EverWallet => {
102                    let mut out_msg = None;
103                    tx.out_msgs
104                        .iterate(|item| {
105                            out_msg = Some(item.0);
106                            Ok(false)
107                        })
108                        .ok()?;
109
110                    let out_msg = out_msg?;
111                    let recipient = match out_msg.header() {
112                        ton_block::CommonMsgInfo::IntMsgInfo(header) => &header.dst,
113                        _ => return None,
114                    };
115
116                    let known_payload = out_msg.body().and_then(parse_payload);
117
118                    (
119                        Some(recipient.clone()),
120                        known_payload,
121                        WalletInteractionMethod::WalletV3Transfer,
122                    )
123                }
124                WalletType::WalletV3R1
125                | WalletType::WalletV3R2
126                | WalletType::WalletV4R1
127                | WalletType::WalletV4R2
128                | WalletType::WalletV5R1 => {
129                    let mut out_msg = None;
130                    tx.out_msgs
131                        .iterate(|item| {
132                            out_msg = Some(item.0);
133                            Ok(false)
134                        })
135                        .ok()?;
136
137                    let out_msg = out_msg?;
138                    let recipient = match out_msg.header() {
139                        ton_block::CommonMsgInfo::IntMsgInfo(header) => &header.dst,
140                        _ => return None,
141                    };
142
143                    let known_payload = out_msg.body().and_then(parse_jetton_payload);
144
145                    (
146                        Some(recipient.clone()),
147                        known_payload,
148                        WalletInteractionMethod::TonWalletTransfer,
149                    )
150                }
151                WalletType::Multisig(multisig_type) => {
152                    let method = parse_multisig_transaction_impl(multisig_type, in_msg, tx)?;
153                    let (recipient, known_payload) = match &method {
154                        MultisigTransaction::Submit(MultisigSubmitTransaction {
155                            payload,
156                            dest,
157                            ..
158                        })
159                        | MultisigTransaction::Send(MultisigSendTransaction {
160                            payload,
161                            dest,
162                            ..
163                        }) => (
164                            Some(dest.clone()),
165                            parse_payload(ton_types::SliceData::load_cell_ref(payload).ok()?),
166                        ),
167                        _ => (None, None),
168                    };
169
170                    (
171                        recipient,
172                        known_payload,
173                        WalletInteractionMethod::Multisig(Box::new(method)),
174                    )
175                }
176            };
177
178            return Some(TransactionAdditionalInfo::WalletInteraction(
179                WalletInteractionInfo {
180                    recipient,
181                    known_payload,
182                    method,
183                },
184            ));
185        }
186        ton_block::CommonMsgInfo::IntMsgInfo(header) => header,
187        ton_block::CommonMsgInfo::ExtOutMsgInfo(_) => return None,
188    };
189
190    let depool_notifications = DePoolParticipantFunctions::instance();
191    let token_notifications = WalletNotificationFunctions::instance();
192
193    if int_header.bounced {
194        return None;
195    }
196
197    let body = in_msg.body()?;
198    let function_id = read_function_id(&body).ok()?;
199
200    if function_id == 0 {
201        parse_comment_payload(body).map(TransactionAdditionalInfo::Comment)
202    } else if function_id == depool_notifications.on_round_complete.input_id {
203        let inputs = depool_notifications
204            .on_round_complete
205            .decode_input(body, true, false)
206            .ok()?;
207
208        DePoolOnRoundCompleteNotification::try_from(InputMessage(inputs))
209            .map(TransactionAdditionalInfo::DePoolOnRoundComplete)
210            .ok()
211    } else if function_id == depool_notifications.receive_answer.input_id {
212        let inputs = depool_notifications
213            .receive_answer
214            .decode_input(body, true, false)
215            .ok()?;
216
217        DePoolReceiveAnswerNotification::try_from(InputMessage(inputs))
218            .map(TransactionAdditionalInfo::DePoolReceiveAnswer)
219            .ok()
220    } else if function_id == token_notifications.notify_wallet_deployed.input_id {
221        let inputs = token_notifications
222            .notify_wallet_deployed
223            .decode_input(body, true, false)
224            .ok()?;
225
226        TokenWalletDeployedNotification::try_from(InputMessage(inputs))
227            .map(TransactionAdditionalInfo::TokenWalletDeployed)
228            .ok()
229    } else if function_id == JettonNotify::FUNCTION_ID {
230        JettonNotify::decode_body(body).map(TransactionAdditionalInfo::JettonNotify)
231    } else {
232        None
233    }
234}

341pub fn parse_multisig_transaction(
342    multisig_type: MultisigType,
343    tx: &ton_block::Transaction,
344) -> Option<MultisigTransaction> {
345    let in_msg = tx.in_msg.as_ref()?.read_struct().ok()?;
346    if !matches!(in_msg.header(), ton_block::CommonMsgInfo::ExtInMsgInfo(_)) {
347        return None;
348    }
349    parse_multisig_transaction_impl(multisig_type, in_msg, tx)
350}
351
352fn parse_multisig_transaction_impl(
353    multisig_type: MultisigType,
354    in_msg: ton_block::Message,
355    tx: &ton_block::Transaction,
356) -> Option<MultisigTransaction> {
357    const PUBKEY_OFFSET: usize = 1 + ed25519_dalek::SIGNATURE_LENGTH * 8 + 1;
358    const PUBKEY_LENGTH: usize = 256;
359    const TIME_LENGTH: usize = 64;
360    const EXPIRE_LENGTH: usize = 32;
361
362    let body = in_msg.body()?;
363    let function_id = {
364        let mut body = body.clone();
365
366        // Shift body by Maybe(signature), Maybe(pubkey), time and expire
367        body.move_by(PUBKEY_OFFSET + PUBKEY_LENGTH + TIME_LENGTH + EXPIRE_LENGTH)
368            .ok()?;
369
370        read_function_id(&body).ok()?
371    };
372
373    let parse_tx_input = |function: &ton_abi::Function,
374                          mut body: ton_types::SliceData|
375     -> Option<(ton_types::UInt256, InputMessage)> {
376        let inputs = function.decode_input(body.clone(), false, false).ok()?;
377        body.move_by(PUBKEY_OFFSET).ok()?;
378        let custodian = body.get_next_hash().ok()?;
379        Some((custodian, InputMessage(inputs)))
380    };
381
382    let parse_tx_full = |function: &ton_abi::Function,
383                         body: ton_types::SliceData|
384     -> Option<(ton_types::UInt256, ContractCall)> {
385        let (custodian, InputMessage(inputs)) = parse_tx_input(function, body)?;
386        let outputs = function.parse(tx).ok()?;
387        Some((custodian, ContractCall { inputs, outputs }))
388    };
389
390    let functions = MultisigFunctions::instance(multisig_type);
391
392    if function_id == functions.send_transaction.input_id {
393        let inputs = functions
394            .send_transaction
395            .decode_input(body, false, false)
396            .ok()?;
397        MultisigSendTransaction::try_from(InputMessage(inputs))
398            .map(MultisigTransaction::Send)
399            .ok()
400    } else if function_id == functions.submit_transaction.input_id {
401        let (custodian, value) = parse_tx_full(functions.submit_transaction, body)?;
402        MultisigSubmitTransaction::try_from((custodian, value))
403            .map(MultisigTransaction::Submit)
404            .ok()
405    } else if function_id == functions.confirm_transaction.input_id {
406        let (custodian, value) = parse_tx_input(functions.confirm_transaction, body)?;
407        MultisigConfirmTransaction::try_from((custodian, value))
408            .map(MultisigTransaction::Confirm)
409            .ok()
410    } else if let Some(functions) = &functions.update_functions {
411        if function_id == functions.submit_update.input_id {
412            let (custodian, value) = parse_tx_full(functions.submit_update, body)?;
413            MultisigSubmitUpdate::try_from((custodian, value))
414                .map(MultisigTransaction::SubmitUpdate)
415                .ok()
416        } else if function_id == functions.confirm_update.input_id {
417            let (custodian, value) = parse_tx_input(functions.confirm_update, body)?;
418            MultisigConfirmUpdate::try_from((custodian, value))
419                .map(MultisigTransaction::ConfirmUpdate)
420                .ok()
421        } else if function_id == functions.execute_update.input_id {
422            let (custodian, value) = parse_tx_input(functions.execute_update, body)?;
423            MultisigExecuteUpdate::try_from((custodian, value))
424                .map(MultisigTransaction::ExecuteUpdate)
425                .ok()
426        } else {
427            None
428        }
429    } else {
430        None
431    }
432}

594pub fn parse_token_transaction(
595    tx: &ton_block::Transaction,
596    description: &ton_block::TransactionDescrOrdinary,
597    version: TokenWalletVersion,
598) -> Option<TokenWalletTransaction> {
599    if description.aborted {
600        return None;
601    }
602
603    let in_msg = tx.in_msg.as_ref()?.read_struct().ok()?;
604
605    let mut body = in_msg.body()?;
606    let function_id = read_function_id(&body).ok()?;
607
608    let header = in_msg.int_header()?;
609
610    let functions = TokenWalletFunctions::for_version(version);
611
612    if header.bounced {
613        body.move_by(32).ok()?;
614        let function_id = read_function_id(&body).ok()?;
615        body.move_by(32).ok()?;
616
617        if function_id == functions.accept_transfer.input_id {
618            return Some(TokenWalletTransaction::TransferBounced(
619                body.get_next_u128().ok()?.into(),
620            ));
621        }
622
623        if function_id == functions.accept_burn.input_id {
624            Some(TokenWalletTransaction::SwapBackBounced(
625                body.get_next_u128().ok()?.into(),
626            ))
627        } else {
628            None
629        }
630    } else if function_id == functions.accept_mint.input_id {
631        let inputs = functions.accept_mint.decode_input(body, true, false).ok()?;
632
633        Accept::try_from((InputMessage(inputs), version))
634            .map(|Accept { tokens }| TokenWalletTransaction::Accept(tokens))
635            .ok()
636    } else if function_id == functions.transfer_to_wallet.input_id {
637        let inputs = functions
638            .transfer_to_wallet
639            .decode_input(body, true, false)
640            .ok()?;
641
642        TokenOutgoingTransfer::try_from((
643            InputMessage(inputs),
644            TransferType::ByTokenWalletAddress,
645            version,
646        ))
647        .map(TokenWalletTransaction::OutgoingTransfer)
648        .ok()
649    } else if function_id == functions.transfer.input_id {
650        let inputs = functions.transfer.decode_input(body, true, false).ok()?;
651
652        TokenOutgoingTransfer::try_from((
653            InputMessage(inputs),
654            TransferType::ByOwnerWalletAddress,
655            version,
656        ))
657        .map(TokenWalletTransaction::OutgoingTransfer)
658        .ok()
659    } else if function_id == functions.accept_transfer.input_id {
660        let inputs = functions
661            .accept_transfer
662            .decode_input(body, true, false)
663            .ok()?;
664
665        TokenIncomingTransfer::try_from((InputMessage(inputs), version))
666            .map(TokenWalletTransaction::IncomingTransfer)
667            .ok()
668    } else if function_id == functions.burn.input_id {
669        let inputs = functions.burn.decode_input(body, true, false).ok()?;
670
671        TokenSwapBack::try_from((InputMessage(inputs), version))
672            .map(TokenWalletTransaction::SwapBack)
673            .ok()
674    } else {
675        None
676    }
677}

679pub fn parse_jetton_transaction(
680    tx: &ton_block::Transaction,
681    description: &ton_block::TransactionDescrOrdinary,
682) -> Option<JettonWalletTransaction> {
683    const STANDART_JETTON_CELLS: usize = 0;
684    const MINTLESS_JETTON_CELLS: usize = 2;
685
686    if description.aborted {
687        return None;
688    }
689
690    let in_msg = tx.in_msg.as_ref()?.read_struct().ok()?;
691
692    // Workaround to extract body since we don`t store the type in message
693    let mut body = {
694        let body = in_msg.body()?;
695        let refs = body.remaining_references();
696
697        match refs {
698            MINTLESS_JETTON_CELLS => {
699                let cell = body.reference_opt(1)?;
700                SliceData::load_cell(cell).ok()?
701            }
702            STANDART_JETTON_CELLS => body,
703            _ => return None,
704        }
705    };
706
707    let opcode = body.get_next_u32().ok()?;
708
709    if opcode != JETTON_TRANSFER_OPCODE && opcode != JETTON_INTERNAL_TRANSFER_OPCODE {
710        return None;
711    }
712
713    // Skip query id
714    body.move_by(64).ok()?;
715
716    let mut grams = ton_block::Grams::default();
717    grams.read_from(&mut body).ok()?;
718
719    let amount = BigUint::from(grams.as_u128());
720
721    let mut addr = MsgAddressInt::default();
722    addr.read_from(&mut body).ok()?;
723
724    match opcode {
725        JETTON_TRANSFER_OPCODE => Some(JettonWalletTransaction::Transfer(JettonOutgoingTransfer {
726            to: addr,
727            tokens: amount,
728        })),
729        JETTON_INTERNAL_TRANSFER_OPCODE => Some(JettonWalletTransaction::InternalTransfer(
730            JettonIncomingTransfer {
731                from: addr,
732                tokens: amount,
733            },
734        )),
735        _ => None,
736    }
737}

748        IDS.get_or_init(|| {
749            Box::new(Self {

758pub fn parse_nft_transaction(
759    tx: &ton_block::Transaction,
760    description: &ton_block::TransactionDescrOrdinary,
761) -> Option<NftTransaction> {
762    if description.aborted {
763        return None;
764    }
765
766    let in_msg = tx.in_msg.as_ref()?.read_struct().ok()?;
767
768    let body = in_msg.body()?;
769    let function_id = read_function_id(&body).ok()?;
770
771    let functions = NftFunctions::instance();
772
773    if function_id == functions.transfer.input_id {
774        let inputs = functions.transfer.decode_input(body, true, false).ok()?;
775
776        IncomingNftTransfer::try_from(InputMessage(inputs))
777            .map(NftTransaction::Transfer)
778            .ok()
779    } else if function_id == functions.change_owner.input_id {
780        let inputs = functions
781            .change_owner
782            .decode_input(body, true, false)
783            .ok()?;
784
785        IncomingChangeOwner::try_from(InputMessage(inputs))
786            .map(NftTransaction::ChangeOwner)
787            .ok()
788    } else if function_id == functions.change_manager.input_id {
789        let inputs = functions
790            .change_manager
791            .decode_input(body, true, false)
792            .ok()?;
793
794        IncomingChangeManager::try_from(InputMessage(inputs))
795            .map(NftTransaction::ChangeManager)
796            .ok()
797    } else {
798        None
799    }
800}
