26    fn get_master_key(&self, master_key: &PublicKey) -> Result<&MasterKey> {
27        match self.master_keys.get(master_key.as_bytes()) {
28            Some(key) => Ok(key),
29            None => Err(MasterKeyError::MasterKeyNotFound.into()),

33    fn use_sign_input(
34        &'_ self,
35        password_cache: &PasswordCache,
36        input: DerivedKeyPassword,
37    ) -> Result<Keypair> {
38        let (master_key, account_id, password) = match input {
39            DerivedKeyPassword::ByAccountId {
40                master_key,
41                account_id,
42                password,
43            } => (self.get_master_key(&master_key)?, account_id, password),
44            DerivedKeyPassword::ByPublicKey {
45                master_key,
46                public_key,
47                password,
48            } => {
49                let master_key = self.get_master_key(&master_key)?;
50                match master_key.accounts_map.get(public_key.as_bytes()) {
51                    Some(account) => (master_key, account.account_id, password),
52                    None => return Err(MasterKeyError::DerivedKeyNotFound.into()),
53                }
54            }
55        };
56
57        let password =
58            password_cache.process_password(master_key.public_key.to_bytes(), password)?;
59
60        let decrypter = ChaCha20Poly1305::new(&symmetric_key_from_password(
61            password.as_ref(),
62            &master_key.salt,
63        ));
64
65        let master = decrypt_secure(
66            &decrypter,
67            &master_key.entropy_nonce,
68            &master_key.enc_entropy,
69        )?;
70
71        let signer = derive_from_master(account_id, master.unsecure())?;
72
73        password.proceed();
74        Ok(signer)
75    }
76}

90    async fn add_key(
91        &mut self,
92        ctx: SignerContext<'_>,
93        input: Self::CreateKeyInput,
94    ) -> Result<SignerEntry> {
95        Ok(match input {
96            DerivedKeyCreateInput::Import {
97                phrase,
98                password,
99                key_name,
100            } => {
101                let (master_key, key_name, password) =
102                    MasterKey::new(ctx.password_cache, password, phrase, key_name)?;
103                let public_key = master_key.public_key;
104
105                match self.master_keys.entry(public_key.to_bytes()) {
106                    hash_map::Entry::Vacant(entry) => {
107                        entry.insert(master_key);
108                    }
109                    hash_map::Entry::Occupied(mut entry) => {
110                        let existing = entry.get_mut();
111                        existing.salt = master_key.salt;
112                        existing.enc_entropy = master_key.enc_entropy;
113                        existing.entropy_nonce = master_key.entropy_nonce;
114                        existing.enc_phrase = master_key.enc_phrase;
115                        existing.phrase_nonce = master_key.phrase_nonce;
116
117                        match existing.accounts_map.entry(public_key.to_bytes()) {
118                            hash_map::Entry::Vacant(entry) => {
119                                entry.insert(Account {
120                                    name: key_name.clone(),
121                                    account_id: 0,
122                                });
123                            }
124                            hash_map::Entry::Occupied(mut entry) => {
125                                entry.get_mut().name = key_name.clone();
126                            }
127                        }
128                    }
129                };
130
131                password.proceed();
132                SignerEntry {
133                    name: key_name,
134                    public_key,
135                    master_key: public_key,
136                    account_id: 0,
137                }
138            }
139            DerivedKeyCreateInput::Derive {
140                master_key,
141                account_id,
142                password,
143                key_name,
144            } => {
145                let master_key = match self.master_keys.get_mut(master_key.as_bytes()) {
146                    Some(key) => key,
147                    None => return Err(MasterKeyError::MasterKeyNotFound.into()),
148                };
149
150                let found = master_key
151                    .accounts_map
152                    .values()
153                    .any(|x| x.account_id == account_id);
154                if found {
155                    return Err(MasterKeyError::DerivedKeyExists.into());
156                }
157
158                let password = ctx
159                    .password_cache
160                    .process_password(master_key.public_key.to_bytes(), password)?;
161
162                let decrypter = ChaCha20Poly1305::new(&symmetric_key_from_password(
163                    password.as_ref(),
164                    &master_key.salt,
165                ));
166
167                let master = decrypt_secure(
168                    &decrypter,
169                    &master_key.entropy_nonce,
170                    &master_key.enc_entropy,
171                )?;
172
173                let public_key = derive_from_master(account_id, master.unsecure())?.public;
174
175                let key_name = key_name.unwrap_or_else(|| default_key_name(public_key.as_bytes()));
176
177                master_key.accounts_map.insert(
178                    public_key.to_bytes(),
179                    Account {
180                        name: key_name.clone(),
181                        account_id,
182                    },
183                );
184
185                password.proceed();
186                SignerEntry {
187                    name: key_name,
188                    public_key,
189                    master_key: master_key.public_key,
190                    account_id,
191                }
192            }
193        })

196    async fn update_key(
197        &mut self,
198        ctx: SignerContext<'_>,
199        input: Self::UpdateKeyInput,
200    ) -> Result<SignerEntry> {
201        match input {
202            Self::UpdateKeyInput::RenameKey {
203                master_key,
204                public_key,
205                name,
206            } => {
207                let entry = match self.master_keys.get_mut(master_key.as_bytes()) {
208                    Some(key) => key,
209                    None => return Err(MasterKeyError::MasterKeyNotFound.into()),
210                };
211
212                let entry = match entry.accounts_map.get_mut(public_key.as_bytes()) {
213                    Some(entry) => entry,
214                    None => return Err(MasterKeyError::DerivedKeyNotFound.into()),
215                };
216
217                entry.name = name.clone();
218
219                Ok(SignerEntry {
220                    name,
221                    public_key,
222                    master_key,
223                    account_id: entry.account_id,
224                })
225            }
226            Self::UpdateKeyInput::ChangePassword {
227                master_key,
228                old_password,
229                new_password,
230            } => {
231                let entry = match self.master_keys.get_mut(master_key.as_bytes()) {
232                    Some(key) => key,
233                    None => return Err(MasterKeyError::MasterKeyNotFound.into()),
234                };
235
236                let old_password = ctx
237                    .password_cache
238                    .process_password(master_key.to_bytes(), old_password)?;
239                let new_password = ctx
240                    .password_cache
241                    .process_password(master_key.to_bytes(), new_password)?;
242
243                entry.change_password(old_password.as_ref(), new_password.as_ref())?;
244
245                let name = entry
246                    .accounts_map
247                    .get(master_key.as_bytes())
248                    .map(|item| item.name.clone())
249                    .unwrap_or_default();
250
251                new_password.proceed();
252                Ok(SignerEntry {
253                    name,
254                    public_key: entry.public_key,
255                    master_key: entry.public_key,
256                    account_id: 0,
257                })
258            }
259        }

369        #[derive(Deserialize)]
370        struct ParsedDerivedKeySigner {
371            master_keys: HashMap<String, MasterKey>,
372        }
373
374        if let Ok(state) = serde_json::from_str::<ParsedDerivedKeySigner>(data) {
375            self.master_keys = state
376                .master_keys
377                .into_iter()
378                .map(|(public_key, master_key)| {
379                    let public_key = hex::decode(public_key)?;
380                    let public_key = PublicKey::from_bytes(&public_key)?;
381                    Result::<_, anyhow::Error>::Ok((public_key.to_bytes(), master_key))
382                })
383                .collect::<Result<HashMap<_, _>, _>>()?;
384        } else {
385            let mut master_keys = HashMap::with_capacity(1);
386            if let Some(master_key) = serde_json::from_str::<Option<MasterKey>>(data)? {
387                master_keys.insert(master_key.public_key.to_bytes(), master_key);
388            }
389            self.master_keys = master_keys;
390        }
391
392        Ok(())
393    }

395    fn store_state(&self) -> String {
396        struct StoredMasterKeys<'a>(&'a HashMap<[u8; 32], MasterKey>);
397
398        impl Serialize for StoredMasterKeys<'_> {
399            fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
400            where
401                S: Serializer,
402            {
403                use serde::ser::SerializeMap;
404
405                let mut map = serializer.serialize_map(Some(self.0.len()))?;
406                for (public_key, master_key) in self.0.iter() {
407                    map.serialize_entry(&hex::encode(public_key), master_key)?;
408                }
409
410                map.end()
411            }
412        }
413
414        #[derive(Serialize)]
415        struct StoredDerivedKeySigner<'a> {
416            master_keys: StoredMasterKeys<'a>,
417        }
418
419        serde_json::to_string(&StoredDerivedKeySigner {
420            master_keys: StoredMasterKeys(&self.master_keys),
421        })
422        .trust_me()
423    }

466struct MasterKey {
467    #[serde(with = "serde_public_key")]
468    public_key: PublicKey,
469
470    #[serde(with = "serde_bytes")]
471    salt: Vec<u8>,
472

474    enc_entropy: Vec<u8>,
475
476    #[serde(with = "serde_nonce")]
477    entropy_nonce: Nonce,
478
479    #[serde(with = "serde_bytes")]
480    enc_phrase: Vec<u8>,
481
482    #[serde(with = "serde_nonce")]
483    phrase_nonce: Nonce,
484
485    #[serde(with = "serde_accounts_map")]
486    accounts_map: AccountsMap,

488
489impl MasterKey {
490    fn new(
491        password_cache: &'_ PasswordCache,
492        password: Password,
493        phrase: SecUtf8,
494        key_name: Option<String>,
495    ) -> Result<(Self, String, PasswordCacheTransaction<'_>)> {
496        use zeroize::Zeroize;
497
498        let mut phrase = phrase.unsecure().to_string();
499
500        // SECURITY: private key will be zeroized here
501        let public_key =
502            derive_from_phrase(&phrase, MnemonicType::Bip39(Bip39MnemonicData::labs_old(0)))?
503                .public;
504
505        let key_name = key_name.unwrap_or_else(|| default_key_name(public_key.as_bytes()));
506
507        let password = password_cache.process_password(public_key.to_bytes(), password)?;
508
509        let mut entropy = labs::derive_master_key(&phrase)?;
510        let EncryptedPart {
511            salt,
512            enc_entropy,
513            entropy_nonce,
514            enc_phrase,
515            phrase_nonce,
516        } = compute_encrypted_part(&entropy, phrase.as_bytes(), password.as_ref())?;
517
518        phrase.zeroize();
519        entropy.zeroize();
520
521        let mut accounts_map = AccountsMap::new();
522        accounts_map.insert(
523            public_key.to_bytes(),
524            Account {
525                name: key_name.clone(),
526                account_id: 0,
527            },
528        );
529
530        Ok((
531            Self {
532                public_key,
533                salt,
534                enc_entropy,
535                entropy_nonce,
536                enc_phrase,
537                phrase_nonce,
538                accounts_map,
539            },
540            key_name,
541            password,
542        ))
543    }
544
545    fn change_password(&mut self, old_password: &str, new_password: &str) -> Result<()> {
546        let decrypter =
547            ChaCha20Poly1305::new(&symmetric_key_from_password(old_password, &self.salt));
548
549        let entropy = decrypt_secure(&decrypter, &self.entropy_nonce, &self.enc_entropy)?;
550        let phrase = decrypt_secure(&decrypter, &self.phrase_nonce, &self.enc_phrase)?;
551
552        let encrypted_part =
553            compute_encrypted_part(entropy.unsecure(), phrase.unsecure(), new_password)?;
554        self.salt = encrypted_part.salt;
555        self.enc_entropy = encrypted_part.enc_entropy;
556        self.entropy_nonce = encrypted_part.entropy_nonce;
557        self.enc_phrase = encrypted_part.enc_phrase;
558        self.phrase_nonce = encrypted_part.phrase_nonce;
559
560        Ok(())
561    }
562}

564fn compute_encrypted_part(entropy: &[u8], phrase: &[u8], password: &str) -> Result<EncryptedPart> {
565    use rand::Rng;
566
567    let rng = &mut rand::thread_rng();
568
569    let mut salt = vec![0u8; 32];
570    rng.fill(salt.as_mut_slice());
571
572    let mut entropy_nonce = [0u8; NONCE_LENGTH];
573    rng.fill(&mut entropy_nonce);
574
575    let mut phrase_nonce = [0u8; NONCE_LENGTH];
576    rng.fill(&mut phrase_nonce);
577
578    let entropy_nonce = Nonce::from(entropy_nonce);
579    let phrase_nonce = Nonce::from(phrase_nonce);
580
581    let encryptor = ChaCha20Poly1305::new(&symmetric_key_from_password(password, &salt));
582    let enc_entropy = encrypt(&encryptor, &entropy_nonce, entropy)?;
583    let enc_phrase = encrypt(&encryptor, &phrase_nonce, phrase)?;
584
585    Ok(EncryptedPart {
586        salt,
587        enc_entropy,
588        entropy_nonce,
589        enc_phrase,
590        phrase_nonce,
591    })
592}

718    pub(super) fn deserialize<'de, D>(deserializer: D) -> Result<AccountsMap, D::Error>
719    where
720        D: serde::Deserializer<'de>,
721    {
722        #[derive(Deserialize)]
723        struct StoredItem {
724            #[serde(default)]
725            name: Option<String>,
726            account_id: u16,
727        }
728
729        let stored_data = HashMap::<String, StoredItem>::deserialize(deserializer)?;
730        stored_data
731            .into_iter()
732            .map(|(public_key, StoredItem { name, account_id })| {
733                let public_key =
734                    hex::decode(public_key)
735                        .map_err(D::Error::custom)
736                        .and_then(|public_key| {
737                            public_key
738                                .try_into()
739                                .map_err(|_| D::Error::custom("Invalid public key"))
740                        })?;
741                let name = name.unwrap_or_else(|| default_key_name(&public_key));
742                Ok((public_key, Account { name, account_id }))
743            })
744            .collect::<Result<_, _>>()
745    }
746}

748fn derive_from_master(id: u16, master: &[u8]) -> Result<ed25519_dalek::Keypair> {
749    use tiny_hderive::bip32;
750
751    let path = format!("m/44'/396'/0'/0/{id}");
752    let key = bip32::ExtendedPrivKey::derive(master, path.as_str())
753        .map_err(|_| MasterKeyError::DerivationError)?
754        .secret();
755
756    let secret =
757        ed25519_dalek::SecretKey::from_bytes(&key).map_err(|_| MasterKeyError::DerivationError)?;
758    let public = ed25519_dalek::PublicKey::from(&secret);
759    Ok(ed25519_dalek::Keypair { secret, public })
760}

762#[derive(Debug, thiserror::Error)]
763enum MasterKeyError {
764    #[error("Master key not found")]
765    MasterKeyNotFound,
766    #[error("Derived key not found")]
767    DerivedKeyNotFound,
768    #[error("Failed to derive account from master key")]
769    DerivationError,
770    #[error("Derived key already exists")]
771    DerivedKeyExists,
772}
